43fn show_trip(
44    (InMut(ui), InMut(tab)): (InMut<Ui>, InMut<TripTab>),
45    trip_q: Query<TripQuery>,
46    entry_q: Query<EntryQuery>,
47    entry_mode_q: Query<(&EntryMode, Option<&EntryEstimate>)>,
48    platform_q: Query<PlatformQuery>,
49    station_q: Query<StationQuery>,
50    mut commands: Commands,
51) {
52    let trip = trip_q.get(tab.trip_entity).unwrap();
53    ui.heading(trip.name.as_str());
54    ui.label(trip.schedule.len().to_string());
55    egui::ScrollArea::vertical().show(ui, |ui| {
56        egui::Grid::new(ui.id().with("lskdfjlsdkjflkdsjf"))
57            .num_columns(3)
58            .striped(true)
59            .show(ui, |ui| {
60                ui.label(tr!("trip-table-station"));
61                ui.label(tr!("trip-table-arrival"));
62                ui.label(tr!("trip-table-departure"));
63                ui.end_row();
64                for it in entry_q.iter_many(trip.schedule.iter()) {
65                    row_ui(
66                        &platform_q,
67                        &station_q,
68                        &entry_mode_q,
69                        ui,
70                        &trip,
71                        &it,
72                        &mut commands,
73                    );
74                    ui.end_row();
75                }
76            });
77    });
78}

80fn row_ui(
81    platform_q: &Query<PlatformQuery>,
82    station_q: &Query<StationQuery>,
83    entry_mode_q: &Query<(&EntryMode, Option<&EntryEstimate>)>,
84    ui: &mut Ui,
85    trip: &TripQueryItem,
86    it: &EntryQueryItem,
87    mut commands: &mut Commands,
88) {
89    const BUTTON_SIZE: Vec2 = vec2(70.0, 18.0);
90    let platform = platform_q.get(it.stop()).unwrap();
91    let station = platform.station(&station_q);
92
93    // display station label
94    ui.label(station.name.as_str());
95
96    // display arrival button
97    let arr_res = match it.mode.arr {
98        None => ui.add_sized(BUTTON_SIZE, egui::Button::new("↓")),
99        Some(TravelMode::Flexible) => ui.add_sized(BUTTON_SIZE, egui::Button::new("〇")),
100        Some(TravelMode::At(t)) => shift_at_value(t, it.entity, ui, commands, BUTTON_SIZE, true),
101        Some(TravelMode::For(d)) => shift_for_value(d, it.entity, ui, commands, BUTTON_SIZE, true),
102    };
103    arrival_popup(
104        &arr_res,
105        &it,
106        &trip,
107        &entry_mode_q,
108        RectAlign::LEFT,
109        &mut commands,
110    );
111
112    // display departure button
113    let dep_res = match it.mode.dep {
114        TravelMode::Flexible => ui.add_sized(BUTTON_SIZE, egui::Button::new("〇")),
115        TravelMode::At(t) => shift_at_value(t, it.entity, ui, commands, BUTTON_SIZE, false),
116        TravelMode::For(d) => shift_for_value(d, it.entity, ui, commands, BUTTON_SIZE, false),
117    };
118    departure_popup(&dep_res, &it, RectAlign::RIGHT, &mut commands);
119}
