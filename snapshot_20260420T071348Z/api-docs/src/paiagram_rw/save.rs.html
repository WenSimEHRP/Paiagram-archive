30pub enum SaveData {
31    CompressedCbor(Vec<u8>),
32    Ron(Vec<u8>),
33}
34
35#[derive(Resource, Deref, DerefMut)]
36pub struct LoadedScene(pub DynamicScene);
37
38pub fn save(scene: DynamicScene, registry: AppTypeRegistry, filename: String) {
39    super::write::write_file(filename, move |writer| {
40        let reg = registry.read();
41        let serializer = SceneSerializer::new(&scene, &reg);
42        let mut encoder = lz4_flex::frame::FrameEncoder::new(writer);
43        cbor4ii::serde::to_writer(&mut encoder, &serializer)
44            .map_err(std::io::Error::other)
45            .and_then(|_| encoder.finish().map(|_| ()).map_err(std::io::Error::other))
46    });
47}
48
49pub fn save_ron(scene: DynamicScene, registry: AppTypeRegistry, filename: String) {
50    super::write::write_file(filename, move |writer| {
51        let reg = registry.read();
52        scene
53            .serialize(&reg)
54            .map_err(std::io::Error::other)
55            .and_then(|s| {
56                writer
57                    .write(s.as_bytes())
58                    .map(|_| ())
59                    .map_err(std::io::Error::other)
60            })
61    });
62}

64fn deserialize_load_candidate(world: &mut World) {
65    let Some(data) = world.remove_resource::<LoadCandidate>() else {
66        error!("Tried to load data but the data does not exist");
67        return;
68    };
69    let registry = world.resource::<AppTypeRegistry>().read();
70    let scene_deserializer = SceneDeserializer {
71        type_registry: &registry,
72    };
73
74    let scene: DynamicScene;
75    match data.0 {
76        SaveData::CompressedCbor(d) => {
77            let mut decoder = lz4_flex::frame::FrameDecoder::new(d.as_slice());
78            let mut deserializer = cbor4ii::serde::Deserializer::new(IoReader::new(
79                std::io::BufReader::new(&mut decoder),
80            ));
81            scene = scene_deserializer.deserialize(&mut deserializer).unwrap();
82        }
83        SaveData::Ron(d) => {
84            let mut deserializer = ron::Deserializer::from_bytes(d.as_slice()).unwrap();
85            scene = scene_deserializer.deserialize(&mut deserializer).unwrap();
86        }
87    }
88    drop(registry);
89    world.insert_resource(LoadedScene(scene));
90}
