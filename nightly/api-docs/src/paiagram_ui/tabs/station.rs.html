20#[derive(MapEntities, Serialize, Deserialize, Clone, PartialEq)]
21pub struct StationTab {
22    #[entities]
23    station_entity: Entity,
24    include_nonstop: bool,
25}

27impl StationTab {
28    pub fn new(station_entity: Entity) -> Self {
29        Self {
30            station_entity,
31            include_nonstop: false,
32        }
33    }
34}

36impl super::Tab for StationTab {
37    const NAME: &'static str = "Station";
38    fn main_display(&mut self, world: &mut World, ui: &mut egui::Ui) {
39        let station_name = world.get::<Name>(self.station_entity).unwrap().as_str();
40        ui.heading(station_name);
41        ui.checkbox(&mut self.include_nonstop, "Include non-stop");
42        egui::ScrollArea::both().show(ui, |ui| {
43            world
44                .run_system_cached_with(display_time_grid, (ui, self))
45                .unwrap();
46        });
47    }
48}

50struct DisplayedEntry<'a> {
51    time: TimetableTime,
52    color: Color32,
53    trip_name: &'a str,
54    last_station_abbrev: &'a str,
55}
56
57fn display_time_grid(
58    (InMut(ui), InRef(tab)): (InMut<egui::Ui>, InRef<StationTab>),
59    station_q: Query<StationQuery>,
60    platform_entry_q: Query<&PlatformEntries>,
61    entry_q: Query<EntryQuery>,
62    class_q: Query<ClassQuery>,
63    trip_q: Query<TripQuery>,
64    parent_station_or_station: Query<ParentStationOrStation>,
65    global_timer: Res<GlobalTimer>,
66) {
67    let mut entry_bucket: [Vec<DisplayedEntry>; 24] = [const { Vec::new() }; 24];
68
69    // Prepare the station info
70    let station_info = station_q.get(tab.station_entity).unwrap();
71    let station_entry_iter = station_info
72        .passing_entries(&platform_entry_q)
73        .map(|e| entry_q.get(e).unwrap());
74    let station_entry_iter = if tab.include_nonstop {
75        Either::Left(station_entry_iter)
76    } else {
77        Either::Right(station_entry_iter.filter(|it| it.mode.arr.is_some()))
78    };
79    for e in station_entry_iter {
80        let Some(estimate) = e.estimate else {
81            continue;
82        };
83        let trip = trip_q.get(e.parent_schedule.parent()).unwrap();
84        let last_entry_entity = *trip.schedule.last().unwrap();
85        let last_stop_entity = entry_q.get(last_entry_entity).unwrap().stop();
86        let last_station_entity = parent_station_or_station
87            .get(last_stop_entity)
88            .unwrap()
89            .parent();
90        let last_station_name = station_q.get(last_station_entity).unwrap().name.as_str();
91        let class = class_q.get(trip.class.entity()).unwrap();
92        entry_bucket[estimate.dep.hour() as usize].push(DisplayedEntry {
93            time: estimate.dep,
94            color: class.stroke.color.get(ui.visuals().dark_mode),
95            trip_name: trip.name.as_str(),
96            last_station_abbrev: last_station_name,
97        });
98    }
99    for line in entry_bucket.iter_mut() {
100        line.sort_by_key(|it| it.time.minute() * 60 + it.time.second());
101    }
102
103    // Display the info
104    let mut heights: [f32; 25] = [0.0; 25];
105    let (current_h, current_min, current_secs, _) =
106        global_timer.read_ticks().to_timetable_time().to_hmsd();
107    let current_h = current_h as usize;
108    let mut widths_seconds: Vec<(f32, i32)> = Vec::with_capacity(entry_bucket[current_h].len() + 2);
109    widths_seconds.push((ui.clip_rect().left(), 0));
110    egui::Grid::new("station grid")
111        .striped(true)
112        .show(ui, |ui| {
113            for (line_idx, entries) in entry_bucket.into_iter().enumerate() {
114                ui.horizontal_centered(|ui| {
115                    ui.heading(line_idx.to_string());
116                });
117                fn display_entry(entry: DisplayedEntry, ui: &mut egui::Ui) {
118                    ui.vertical_centered(|ui| {
119                        ui.small(entry.last_station_abbrev);
120                        ui.label(
121                            RichText::new(entry.time.minute().to_string())
122                                .color(entry.color)
123                                .font(egui::FontId::new(
124                                    16.0,
125                                    egui::FontFamily::Name("dia_pro".into()),
126                                )),
127                        );
128                        ui.small(entry.trip_name);
129                    });
130                }
131                if line_idx == current_h {
132                    let mut push_widths_seconds = |minutes: i32, ui: &egui::Ui| {
133                        if widths_seconds.last().unwrap().1 == minutes * 60 {
134                            return;
135                        }
136                        widths_seconds.push((ui.cursor().left(), minutes * 60));
137                    };
138                    for entry in entries {
139                        push_widths_seconds(entry.time.minute(), ui);
140                        display_entry(entry, ui);
141                    }
142                    push_widths_seconds(60, ui);
143                } else {
144                    for entry in entries {
145                        display_entry(entry, ui);
146                    }
147                }
148                heights[line_idx] = ui.cursor().top();
149                ui.end_row();
150            }
151        });
152    heights[24] = ui.cursor().top();
153
154    // Calculate the time indicators' position
155    let current_seconds = current_min * 60 + current_secs;
156    let hour_progress = current_seconds as f32 / 3600.0;
157    let base_y = heights[current_h];
158    let next_y = heights[current_h + 1];
159    let block_height = next_y - base_y;
160    let mut hour_line_y = base_y + block_height * hour_progress;
161    let (base_x_idx, (base_x, base_seconds)) = widths_seconds
162        .iter()
163        .enumerate()
164        .rev()
165        .find(|(_, (_, secs))| current_seconds >= *secs)
166        .unwrap();
167    let (next_x, next_seconds) = widths_seconds[base_x_idx + 1];
168    let width = next_x - base_x;
169    let width_seconds = next_seconds - base_seconds;
170    let minute_progress = (current_seconds - base_seconds) as f32 / (width_seconds) as f32;
171    let mut minute_line_x = base_x + width * minute_progress;
172    let line_stroke = Stroke::new(1.5, Color32::RED);
173    line_stroke.round_center_to_pixel(ui.pixels_per_point(), &mut hour_line_y);
174    line_stroke.round_center_to_pixel(ui.pixels_per_point(), &mut minute_line_x);
175
176    // Draw the indicators
177    display_time_indicator_indicator_vertical(
178        ui.id().with("hour indicator"),
179        ui.clip_rect(),
180        hour_line_y,
181        line_stroke.color,
182        ui.painter(),
183    );
184    ui.painter().hline(
185        ui.clip_rect().left()..=ui.clip_rect().right(),
186        hour_line_y,
187        line_stroke,
188    );
189    display_time_indicator_indicator_horizontal(
190        ui.id().with("minute indicator"),
191        Rect::from_x_y_ranges(ui.clip_rect().x_range(), base_y..=next_y),
192        minute_line_x,
193        line_stroke.color,
194        ui.painter(),
195    );
196    ui.painter()
197        .vline(minute_line_x, base_y..=next_y, line_stroke);
198}
