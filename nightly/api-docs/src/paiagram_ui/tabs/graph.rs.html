32fn segment_visible(viewport: Rect, a: egui::Pos2, b: egui::Pos2) -> bool {
33    if viewport.contains(a) || viewport.contains(b) {
34        return true;
35    }
36    let min_x = a.x.min(b.x);
37    let max_x = a.x.max(b.x);
38    let min_y = a.y.min(b.y);
39    let max_y = a.y.max(b.y);
40    let seg_rect = Rect::from_min_max(egui::pos2(min_x, min_y), egui::pos2(max_x, max_y));
41    viewport.intersects(seg_rect)
42}

222fn display(tab: &mut GraphTab, world: &mut World, ui: &mut egui::Ui) {
223    let (response, mut painter) =
224        ui.allocate_painter(ui.available_size_before_wrap(), Sense::click_and_drag());
225    tab.navi.visible = response.rect;
226    tab.navi.handle_navigation(ui, &response);
227    let attribution = world
228        .run_system_cached_with(
229            underlay::draw_underlay,
230            (&mut painter, &tab.navi, ui, tab.underlay_tile_change),
231        )
232        .unwrap();
233    let mut state = tab.gpu_state.lock();
234    if let Some(target_format) = ui.ctx().data(|data| {
235        data.get_temp::<eframe::egui_wgpu::wgpu::TextureFormat>(egui::Id::new("wgpu_target_format"))
236    }) {
237        state.target_format = Some(target_format);
238    }
239    if let Some(msaa_samples) = ui
240        .ctx()
241        .data(|data| data.get_temp::<u32>(egui::Id::new("wgpu_msaa_samples")))
242    {
243        state.msaa_samples = msaa_samples;
244    }
245    let interact_pos = response
246        .clicked()
247        .then_some(ui.input(|r| r.pointer.interact_pos()))
248        .flatten();
249    let (selected_stations, selected_trips) = {
250        let selected_items = world.resource::<SelectedItems>();
251        let station_set: Vec<_> = selected_items
252            .station_selection()
253            .iter()
254            .map(|it| it.station)
255            .collect();
256        let mut trip_set: Vec<_> = selected_items
257            .entry_selection()
258            .iter()
259            .map(|it| it.parent)
260            .collect();
261        trip_set.sort_unstable();
262        trip_set.dedup();
263        (station_set, trip_set)
264    };
265    let selected_item = world
266        .run_system_cached_with(
267            push_draw_items,
268            (
269                ui.visuals().dark_mode,
270                &tab.navi,
271                &mut state.instances,
272                &mut painter,
273                interact_pos,
274                &selected_stations,
275                &selected_trips,
276                ui.ctx()
277                    .animate_bool(ui.id().with("gugugaga"), tab.navi.zoom > 0.002),
278            ),
279        )
280        .unwrap();
281    let callback = gpu_draw::paint_callback(response.rect, tab.gpu_state.clone());
282    painter.add(callback);
283
284    if let Some(attribution) = attribution {
285        draw_attribution(ui, response.rect, &attribution);
286    }
287    draw_scale_bar(
288        &painter,
289        response.rect,
290        tab.navi.zoom,
291        ui.visuals().text_color(),
292    );
293
294    // handle selection
295    let selected_items = world.resource_mut::<SelectedItems>().into_inner();
296    let shift_pressed = ui.input(|i| i.modifiers.shift);
297    if shift_pressed
298        && let SelectedItems::Stations(stations) = selected_items
299        && stations.len() == 1
300        && let Some(hover_pos) = ui.input(|r| r.pointer.hover_pos())
301    {
302        let entity = stations[0].station;
303        let (x, y) = world.get::<Node>(entity).unwrap().pos.to_xy();
304        let pos = tab.navi.xy_to_screen_pos(x, y);
305        ui.painter()
306            .line_segment([pos, hover_pos], Stroke::new(1.0, Color32::BLUE));
307    }
308    let selected_items = world.resource::<SelectedItems>();
309    if let SelectedItems::Stations(stations) = selected_items {
310        for station in stations.clone().iter() {
311            let coor = world.get::<Node>(station.station).unwrap().pos;
312            let (x, y) = coor.to_xy();
313            let pos = tab.navi.xy_to_screen_pos(x, y);
314            let rect = Rect::from_pos(pos).expand(8.0);
315            let res = ui
316                .interact(
317                    rect,
318                    ui.id().with(station.station).with("popup response"),
319                    Sense::drag(),
320                )
321                .on_hover_cursor(egui::CursorIcon::Grab);
322            if res.dragged() {
323                ui.ctx().set_cursor_icon(egui::CursorIcon::Grabbing);
324                let new_pos = pos + res.drag_delta();
325                let (x, y) = tab.navi.screen_pos_to_xy(new_pos);
326                let new_coor = NodePos::from_xy(x, y);
327                world.get_mut::<Node>(station.station).unwrap().pos = new_coor;
328            }
329            let inner = |ui: &mut Ui| {
330                ui.set_width(150.0);
331                world.get_mut::<Name>(station.station).unwrap().mutate(|s| {
332                    ui.text_edit_singleline(s);
333                });
334                ui.small(coor.to_string());
335            };
336            egui::Popup::menu(&res)
337                .open_memory(Some(egui::SetOpenCommand::Bool(true)))
338                .close_behavior(egui::PopupCloseBehavior::CloseOnClickOutside)
339                .show(inner);
340        }
341    }
342    let selected_items = world.resource_mut::<SelectedItems>().into_inner();
343    // TODO: fix the lifetime here
344    match (selected_item, selected_items) {
345        (Some(SelectedItem::Stations(station)), SelectedItems::Stations(stations))
346            if shift_pressed && stations.len() == 1 =>
347        {
348            let prev_station = stations[0];
349            stations[0] = station;
350            world.trigger(AddIntervalPair {
351                source: prev_station.station,
352                target: station.station,
353                length: Distance::from_m(1000),
354            });
355        }
356        (Some(item), items) => {
357            let ctrl_pressed = ui.input(|i| i.modifiers.ctrl || i.modifiers.command);
358            if ctrl_pressed {
359                items.add_entry(item);
360            } else {
361                items.set_or_reset(item);
362            }
363        }
364        (None, _) => {}
365    }
366    // create new station
367    if response.secondary_clicked()
368        && let Some(pos) = ui.input(|r| r.pointer.interact_pos())
369    {
370        tab.clicked_coor = Some(tab.navi.screen_pos_to_xy(pos));
371    } else if response.clicked() {
372        tab.clicked_coor = None
373    }
374    if let Some((x, y)) = tab.clicked_coor {
375        let pos = tab.navi.xy_to_screen_pos(x, y);
376        let rect = Rect::from_pos(pos).expand(8.0);
377        ui.painter()
378            .circle_filled(pos, 6.0, PredefinedColor::Red.get(ui.visuals().dark_mode));
379        let res = ui
380            .interact(rect, ui.id().with("popup response"), Sense::drag())
381            .on_hover_cursor(egui::CursorIcon::Grab);
382        if res.dragged() {
383            ui.ctx().set_cursor_icon(egui::CursorIcon::Grabbing);
384            let new_pos = pos + res.drag_delta();
385            tab.clicked_coor = Some(tab.navi.screen_pos_to_xy(new_pos));
386        }
387        let inner = |ui: &mut Ui| {
388            ui.set_width(200.0);
389            ui.text_edit_singleline(&mut tab.new_station_name);
390            let coor = NodePos::from_xy(x, y);
391            if ui.button("New Station").clicked() {
392                tab.clicked_coor = None;
393                let name = if tab.new_station_name.is_empty() {
394                    None
395                } else {
396                    Some(tab.new_station_name.clone())
397                };
398                world.trigger(CreateNewStation { name, pos: coor });
399                tab.new_station_name.clear();
400            }
401            ui.small(coor.to_string());
402        };
403        egui::Popup::menu(&res)
404            .open_memory(Some(egui::SetOpenCommand::Bool(true)))
405            .close_behavior(egui::PopupCloseBehavior::CloseOnClickOutside)
406            .show(inner);
407    }
408}

410fn draw_scale_bar(painter: &Painter, viewport: Rect, zoom: f32, color: egui::Color32) {
411    if zoom <= 0.0 || !viewport.is_positive() {
412        return;
413    }
414
415    let desired_px = 120.0f64;
416    let meters_per_px = 1.0 / zoom as f64;
417    let raw_meters = desired_px * meters_per_px;
418    let bar_meters = round_to_1_2_5(raw_meters).max(1.0);
419    let bar_px = (bar_meters as f32 * zoom).max(1.0);
420
421    let margin = 10.0;
422    let baseline_y = viewport.bottom() - margin;
423    let left_x = viewport.left() + margin;
424    let right_x = left_x + bar_px;
425
426    let stroke = Stroke::new(1.6, color);
427    painter.line_segment(
428        [
429            Pos2::new(left_x, baseline_y),
430            Pos2::new(right_x, baseline_y),
431        ],
432        stroke,
433    );
434
435    let tick_len = 7.0;
436    painter.line_segment(
437        [
438            Pos2::new(left_x, baseline_y),
439            Pos2::new(left_x, baseline_y - tick_len),
440        ],
441        stroke,
442    );
443    painter.line_segment(
444        [
445            Pos2::new(right_x, baseline_y),
446            Pos2::new(right_x, baseline_y - tick_len),
447        ],
448        stroke,
449    );
450
451    let mid_tick_len = 5.0;
452    for fraction in [0.25f32, 0.5, 0.75] {
453        let x = left_x + bar_px * fraction;
454        painter.line_segment(
455            [
456                Pos2::new(x, baseline_y),
457                Pos2::new(x, baseline_y - mid_tick_len),
458            ],
459            stroke,
460        );
461    }
462
463    painter.text(
464        Pos2::new(left_x, baseline_y - tick_len - 3.0),
465        Align2::LEFT_BOTTOM,
466        format_scale_label(bar_meters),
467        FontId::proportional(12.0),
468        color,
469    );
470}

472fn round_to_1_2_5(value: f64) -> f64 {
473    if value <= 0.0 {
474        return 0.0;
475    }
476    let exponent = value.log10().floor();
477    let base = 10.0f64.powf(exponent);
478    let normalized = value / base;
479    let rounded = if normalized <= 1.0 {
480        1.0
481    } else if normalized <= 2.0 {
482        2.0
483    } else if normalized <= 5.0 {
484        5.0
485    } else {
486        10.0
487    };
488    rounded * base
489}

491fn format_scale_label(meters: f64) -> String {
492    if meters >= 1000.0 {
493        let km = meters / 1000.0;
494        if (km - km.round()).abs() < 1e-6 {
495            format!("{:.0} km", km)
496        } else {
497            format!("{:.1} km", km)
498        }
499    } else {
500        format!("{:.0} m", meters)
501    }
502}

504fn draw_attribution(ui: &mut Ui, viewport: Rect, attribution: &Attribution) {
505    let margin = 6.0;
506    let font_id = FontId::proportional(13.0);
507    let color = ui.style().visuals.hyperlink_color;
508    let text = format!("© {}", attribution.text);
509    let galley = ui.painter().layout_no_wrap(text.clone(), font_id, color);
510    let size = galley.size();
511    let min = Pos2::new(
512        viewport.right() - margin - size.x,
513        viewport.bottom() - margin - size.y,
514    );
515    let rect = Rect::from_min_size(min, size);
516    let mut r = CornerRadius::ZERO;
517    r.nw = 4;
518    ui.painter()
519        .rect_filled(rect.expand(margin), r, Color32::WHITE.gamma_multiply(0.5));
520    ui.put(
521        rect,
522        egui::Hyperlink::from_label_and_url(text, attribution.url).open_in_new_tab(true),
523    );
524}

526fn push_draw_items(
527    (
528        In(is_dark),
529        InRef(navi),
530        InMut(buffer),
531        InMut(painter),
532        In(maybe_interact_pos),
533        InRef(selected_stations),
534        InRef(selected_trips),
535        In(text_strength),
536    ): (
537        In<bool>,
538        InRef<GraphNavigation>,
539        InMut<Vec<ShapeInstance>>,
540        InMut<Painter>,
541        In<Option<Pos2>>,
542        InRef<[Entity]>,
543        InRef<[Entity]>,
544        In<f32>,
545    ),
546    nodes: Query<(Entity, &Node, Option<&Name>)>,
547    spatial_index: Res<GraphSpatialIndex>,
548    interval_spatial_index: Res<GraphIntervalSpatialIndex>,
549    trip_spatial_index: Res<TripSpatialIndex>,
550    settings: Res<ProjectSettings>,
551    trip_meta_q: Query<(&Name, &TripClass), With<Trip>>,
552    stroke_q: Query<&DisplayedStroke, With<Class>>,
553    timer: Res<GlobalTimer>,
554) -> Option<SelectedItem> {
555    buffer.clear();
556    let mut selected = SelectedItem::None;
557    let time = timer.read_seconds();
558    let repeat_time = settings.repeat_frequency.0 as f64;
559    let query_time = if repeat_time > 0.0 {
560        time.rem_euclid(repeat_time)
561    } else {
562        time
563    };
564    let color = PredefinedColor::Neutral.get(is_dark);
565    let view = navi.visible_rect();
566    let view_expanded = view.expand2(Vec2::splat(12.0));
567    let margin_x = 12.0 / navi.zoom_x().max(f32::EPSILON) as f64;
568    let margin_y = 12.0 / navi.zoom_y().max(f32::EPSILON) as f64;
569    let visible_x = navi.visible_x();
570    let visible_y = navi.visible_y();
571    let min_x = visible_x.start - margin_x;
572    let max_x = visible_x.end + margin_x;
573    let min_y = visible_y.start - margin_y;
574    let max_y = visible_y.end + margin_y;
575
576    const STATION_SELECTION_RADIUS: f32 = 10.0;
577    const SELECTION_RADIUS: f32 = 10.0;
578
579    let candidate_nodes: Vec<Entity> =
580        spatial_index.entities_in_xy_aabb(min_x, min_y, max_x, max_y);
581
582    for entity in candidate_nodes {
583        let Ok((_, node, name)) = nodes.get(entity) else {
584            continue;
585        };
586        let [x, y] = node.pos.to_xy_arr();
587        let pos = navi.xy_to_screen_pos(x, y);
588
589        if let Some(interact_pos) = maybe_interact_pos
590            && selected.is_none()
591        {
592            let r = Rect::from_pos(pos).expand(STATION_SELECTION_RADIUS);
593            if r.contains(interact_pos) {
594                selected = SelectedItem::Stations(StationSelection { station: entity })
595            };
596        }
597
598        if selected_stations.contains(&entity) {
599            painter.circle(
600                pos,
601                SELECTION_RADIUS,
602                Color32::RED.gamma_multiply(0.5),
603                Stroke::new(1.0, Color32::RED),
604            );
605        }
606
607        if !view_expanded.contains(pos) {
608            continue;
609        }
610
611        buffer.push(gpu_draw::ShapeInstance::circle(pos, 4.0, color));
612        if text_strength > 0.05
613            && let Some(name) = name
614        {
615            painter.text(
616                pos + Vec2 { x: 7.0, y: 0.0 },
617                Align2::LEFT_CENTER,
618                name.as_ref(),
619                FontId::proportional(13.0),
620                color.gamma_multiply(text_strength),
621            );
622        }
623    }
624
625    let mut rendered_intervals: HashSet<Entity> = HashSet::new();
626
627    // TODO: interval selection
628    for segment in interval_spatial_index.query_xy_aabb(min_x, min_y, max_x, max_y) {
629        if !rendered_intervals.insert(segment.interval) {
630            continue;
631        }
632        let spos = navi.xy_to_screen_pos(segment.p0[0], segment.p0[1]);
633        let tpos = navi.xy_to_screen_pos(segment.p1[0], segment.p1[1]);
634        if !segment_visible(view_expanded, spos, tpos) {
635            continue;
636        }
637        buffer.push(gpu_draw::ShapeInstance::segment(spos, tpos, 1.0, color));
638    }
639
640    for sample in
641        trip_spatial_index.query_xy_time(min_x..=max_x, min_y..=max_y, query_time..=query_time)
642    {
643        let Ok((name, trip_class)) = trip_meta_q.get(sample.trip) else {
644            continue;
645        };
646        let Ok(stroke) = stroke_q.get(trip_class.entity()) else {
647            continue;
648        };
649        let color = stroke.color.get(is_dark);
650        let pos0 = navi.xy_to_screen_pos(sample.p0[0], sample.p0[1]);
651        let pos1 = navi.xy_to_screen_pos(sample.p1[0], sample.p1[1]);
652        let pos = if query_time <= sample.t1 {
653            pos0
654        } else if query_time >= sample.t2 {
655            pos1
656        } else {
657            let f = (query_time - sample.t1) / (sample.t2 - sample.t1).max(f64::EPSILON);
658            pos0.lerp(pos1, f as f32)
659        };
660
661        if let Some(interact_pos) = maybe_interact_pos
662            && selected.is_none()
663        {
664            let r = Rect::from_pos(pos).expand(STATION_SELECTION_RADIUS);
665            if r.contains(interact_pos) {
666                selected = SelectedItem::TimetableEntries(TimetableEntrySelection {
667                    entry: sample.entry1,
668                    parent: sample.trip,
669                })
670            };
671        }
672
673        if selected_trips.contains(&sample.trip) {
674            painter.circle(
675                pos,
676                SELECTION_RADIUS,
677                Color32::BLUE.gamma_multiply(0.5),
678                Stroke::new(1.0, Color32::BLUE),
679            );
680        }
681
682        buffer.push(gpu_draw::ShapeInstance::stealth_arrow(
683            pos0, pos1, pos, color,
684        ));
685        if text_strength > 0.05 {
686            painter.text(
687                pos + Vec2 { x: 7.0, y: 0.0 },
688                Align2::LEFT_CENTER,
689                name.as_ref(),
690                FontId::proportional(13.0),
691                color.gamma_multiply(text_strength),
692            );
693        }
694    }
695    maybe_interact_pos.map(|_| selected)
696}
