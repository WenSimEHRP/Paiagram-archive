104pub fn draw_underlay(
105    (InMut(painter), InRef(navi), InMut(ui), In(new_type)): (
106        InMut<Painter>,
107        InRef<super::GraphNavigation>,
108        InMut<Ui>,
109        In<Option<UnderlayTileType>>,
110    ),
111    mut visited: Local<HashSet<TileId>>,
112    mut stack: Local<Vec<TileId>>,
113    mut tiles: Local<Option<Option<HttpTiles>>>,
114) -> Option<Attribution> {
115    let text_color = painter.ctx().style().visuals.text_color();
116
117    draw_world_grid(
118        &painter,
119        navi.visible_rect(),
120        navi.offset_x() as f32,
121        navi.offset_y() as f32,
122        navi.zoom_x(),
123    );
124
125    let tiles = tiles.get_or_insert(None);
126
127    let ctx = ui.ctx().clone();
128
129    match new_type {
130        None => {}
131        Some(UnderlayTileType::None) => *tiles = None,
132        Some(UnderlayTileType::OpenStreetMap) => *tiles = Some(HttpTiles::new(OpenStreetMap, ctx)),
133        Some(UnderlayTileType::ChiriinChizu) => *tiles = Some(HttpTiles::new(ChiriinChizu, ctx)),
134        Some(UnderlayTileType::AutoNavi) => *tiles = Some(HttpTiles::new(AutoNavi, ctx)),
135    }
136
137    let Some(tiles) = tiles else {
138        return None;
139    };
140
141    let graph_zoom = navi.zoom_x() as f64;
142    let tile_zoom = graph_zoom_to_tile_zoom(graph_zoom);
143
144    let view = navi.visible_rect();
145
146    let center_screen = view.center();
147    let (center_x, center_y) = navi.screen_pos_to_xy(center_screen);
148    let (center_lon, center_lat) = xy_to_lon_lat(center_x, center_y);
149    let map_center = walkers::lon_lat(center_lon, center_lat);
150    let map_center_projected = mercator::project(map_center, tile_zoom);
151
152    visited.clear();
153    stack.clear();
154    stack.push(root_tile_id(
155        map_center_projected.x(),
156        map_center_projected.y(),
157        tile_zoom,
158        tiles.tile_size(),
159    ));
160    let corrected_tile_size = corrected_tile_size(tiles.tile_size(), tile_zoom);
161    let clip = painter.clip_rect();
162
163    while let Some(tile_id) = stack.pop() {
164        if !visited.insert(tile_id) {
165            continue;
166        }
167
168        let tile_rect = tile_rect(
169            tile_id,
170            corrected_tile_size,
171            map_center_projected.x(),
172            map_center_projected.y(),
173            clip,
174        );
175        if !clip.intersects(tile_rect) {
176            continue;
177        }
178
179        if let Some(tile_piece) = tiles.at(tile_id) {
180            draw_tile_piece(&painter, tile_piece, tile_rect);
181        }
182
183        for next in [
184            tile_id.north(),
185            tile_id.east(),
186            tile_id.south(),
187            tile_id.west(),
188        ]
189        .into_iter()
190        .flatten()
191        {
192            if !visited.contains(&next) {
193                stack.push(next);
194            }
195        }
196    }
197    Some(tiles.attribution())
198}

200fn draw_world_grid(painter: &Painter, viewport: Rect, offset_x: f32, offset_y: f32, zoom: f32) {
201    if zoom <= 0.0 {
202        return;
203    }
204
205    const MIN_WIDTH: f32 = 32.0;
206    const MAX_WIDTH: f32 = 120.0;
207    let base_color = egui::Color32::from_gray(160);
208
209    for p in ((-5)..=5).rev() {
210        let spacing = 10.0f32.powi(p);
211        let screen_spacing = spacing * zoom;
212        let strength =
213            ((screen_spacing * 1.5 - MIN_WIDTH) / (MAX_WIDTH - MIN_WIDTH)).clamp(0.0, 1.0);
214        if strength <= 0.0 {
215            continue;
216        }
217
218        let stroke = Stroke::new(0.6, base_color.gamma_multiply(strength));
219
220        let mut n = (offset_x / spacing).floor();
221        loop {
222            let world_x = n * spacing;
223            let screen_x_rel = (world_x - offset_x) * zoom;
224            if screen_x_rel > viewport.width() {
225                break;
226            }
227            if screen_x_rel >= 0.0 {
228                painter.vline(viewport.left() + screen_x_rel, viewport.y_range(), stroke);
229            }
230            n += 1.0;
231        }
232
233        let mut m = (offset_y / spacing).floor();
234        loop {
235            let world_y = m * spacing;
236            let screen_y_rel = (world_y - offset_y) * zoom;
237            if screen_y_rel > viewport.height() {
238                break;
239            }
240            if screen_y_rel >= 0.0 {
241                painter.hline(viewport.x_range(), viewport.top() + screen_y_rel, stroke);
242            }
243            m += 1.0;
244        }
245    }
246}

248fn graph_zoom_to_tile_zoom(graph_zoom: f64) -> f64 {
249    const TILE_SIZE: f64 = 256.0;
250    let (x0, _) = lon_lat_to_xy(-180.0, 0.0);
251    let (x1, _) = lon_lat_to_xy(180.0, 0.0);
252    let world_meters = (x1 - x0).abs();
253    (graph_zoom * world_meters / TILE_SIZE)
254        .log2()
255        .clamp(0.0, 26.0)
256}

258fn corrected_tile_size(source_tile_size: u32, zoom: f64) -> f64 {
259    source_tile_size as f64 * 2f64.powf(zoom - zoom.round())
260}
261
262fn root_tile_id(center_px_x: f64, center_px_y: f64, zoom: f64, source_tile_size: u32) -> TileId {
263    let rounded_zoom = zoom.round().clamp(0.0, 26.0) as i32;
264    let zoom_offset = ((source_tile_size as f64) / 256.0).log2() as i32;
265    let tile_zoom = (rounded_zoom - zoom_offset).clamp(0, 26) as u8;
266    let tile_size_px = corrected_tile_size(source_tile_size, zoom);
267    let max_index = (2u32.pow(tile_zoom as u32).saturating_sub(1)) as i64;
268
269    let x = ((center_px_x / tile_size_px).floor() as i64).clamp(0, max_index) as u32;
270    let y = ((center_px_y / tile_size_px).floor() as i64).clamp(0, max_index) as u32;
271
272    TileId {
273        x,
274        y,
275        zoom: tile_zoom,
276    }
277}

279fn tile_rect(
280    tile_id: TileId,
281    corrected_tile_size: f64,
282    center_px_x: f64,
283    center_px_y: f64,
284    clip: Rect,
285) -> Rect {
286    let tile_px_x = tile_id.x as f64 * corrected_tile_size;
287    let tile_px_y = tile_id.y as f64 * corrected_tile_size;
288
289    let screen_x = clip.center().x as f64 + (tile_px_x - center_px_x);
290    let screen_y = clip.center().y as f64 + (tile_px_y - center_px_y);
291
292    Rect::from_min_max(
293        pos2(screen_x as f32, screen_y as f32),
294        pos2(
295            (screen_x + corrected_tile_size) as f32,
296            (screen_y + corrected_tile_size) as f32,
297        ),
298    )
299}

302fn draw_tile_piece(painter: &Painter, tile_piece: walkers::TilePiece, rect: Rect) {
303    match tile_piece.tile {
304        Tile::Raster(texture_handle) => {
305            let mut mesh = Mesh::with_texture(texture_handle.id());
306            mesh.add_rect_with_uv(rect, tile_piece.uv, egui::Color32::WHITE);
307            painter.add(Shape::mesh(mesh));
308        }
309        _ => {}
310    }
