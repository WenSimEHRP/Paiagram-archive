9        let rgba = color.to_array();
10        Self {
11            a: [a.x, a.y],
12            b: [b.x, b.y],
13            size: width,
14            color: [
15                rgba[0] as f32 / 255.0,

28            size: radius,
29            color: [

39    pub fn stealth_arrow(from: Pos2, to: Pos2, center: Pos2, color: Color32) -> Self {
40        let direction = to - from;
41        let direction_len = direction.length();
42        let unit_direction = if direction_len > f32::EPSILON {
43            direction / direction_len
44        } else {
45            Vec2::X
46        };
47        let rgba = color.to_array();
48        Self {
49            a: [center.x, center.y],
50            b: [center.x + unit_direction.x, center.y + unit_direction.y],
51            size: 14.0,
52            color: [
53                rgba[0] as f32 / 255.0,
54                rgba[1] as f32 / 255.0,
55                rgba[2] as f32 / 255.0,
56                rgba[3] as f32 / 255.0,
57            ],
58            kind: 2,
59        }
60    }

97struct GraphRenderResources {
98    pipeline: wgpu::RenderPipeline,
99    bind_group: wgpu::BindGroup,
100    uniform_buffer: wgpu::Buffer,
101    vertex_buffer: wgpu::Buffer,
102    vertex_capacity: usize,
103    target_format: wgpu::TextureFormat,
104    msaa_samples: u32,
105}

107impl GraphRenderResources {
108    fn new(device: &wgpu::Device, target_format: wgpu::TextureFormat, msaa_samples: u32) -> Self {
109        let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
110            label: Some("gpu_graph_shader"),
111            source: wgpu::ShaderSource::Wgsl(include_str!("gpu_graph.wgsl").into()),
112        });
113
114        let uniform_buffer = device.create_buffer(&wgpu::BufferDescriptor {
115            label: Some("gpu_graph_uniform"),
116            size: (std::mem::size_of::<[f32; 4]>() as u64).max(16),
117            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
118            mapped_at_creation: false,
119        });
120
121        let bind_group_layout = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
122            label: Some("gpu_graph_bind_group_layout"),
123            entries: &[wgpu::BindGroupLayoutEntry {
124                binding: 0,
125                visibility: wgpu::ShaderStages::VERTEX,
126                ty: wgpu::BindingType::Buffer {
127                    ty: wgpu::BufferBindingType::Uniform,
128                    has_dynamic_offset: false,
129                    min_binding_size: None,
130                },
131                count: None,
132            }],
133        });
134
135        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
136            label: Some("gpu_graph_bind_group"),
137            layout: &bind_group_layout,
138            entries: &[wgpu::BindGroupEntry {
139                binding: 0,
140                resource: uniform_buffer.as_entire_binding(),
141            }],
142        });
143
144        let pipeline_layout = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
145            label: Some("gpu_graph_pipeline_layout"),
146            bind_group_layouts: &[&bind_group_layout],
147            push_constant_ranges: &[],
148        });
149
150        let pipeline = device.create_render_pipeline(&wgpu::RenderPipelineDescriptor {
151            label: Some("gpu_graph_pipeline"),
152            layout: Some(&pipeline_layout),
153            vertex: wgpu::VertexState {
154                module: &shader,
155                entry_point: Some("vs_main"),
156                buffers: &[wgpu::VertexBufferLayout {
157                    array_stride: std::mem::size_of::<ShapeInstance>() as u64,
158                    step_mode: wgpu::VertexStepMode::Instance,
159                    attributes: &[
160                        wgpu::VertexAttribute {
161                            format: wgpu::VertexFormat::Float32x2,
162                            offset: 0,
163                            shader_location: 0,
164                        },
165                        wgpu::VertexAttribute {
166                            format: wgpu::VertexFormat::Float32x2,
167                            offset: (2 * std::mem::size_of::<f32>()) as u64,
168                            shader_location: 1,
169                        },
170                        wgpu::VertexAttribute {
171                            format: wgpu::VertexFormat::Float32,
172                            offset: (4 * std::mem::size_of::<f32>()) as u64,
173                            shader_location: 2,
174                        },
175                        wgpu::VertexAttribute {
176                            format: wgpu::VertexFormat::Float32x4,
177                            offset: (5 * std::mem::size_of::<f32>()) as u64,
178                            shader_location: 3,
179                        },
180                        wgpu::VertexAttribute {
181                            format: wgpu::VertexFormat::Uint32,
182                            offset: (9 * std::mem::size_of::<f32>()) as u64,
183                            shader_location: 4,
184                        },
185                    ],
186                }],
187                compilation_options: wgpu::PipelineCompilationOptions::default(),
188            },
189            fragment: Some(wgpu::FragmentState {
190                module: &shader,
191                entry_point: Some("fs_main"),
192                targets: &[Some(wgpu::ColorTargetState {
193                    format: target_format,
194                    blend: Some(wgpu::BlendState::ALPHA_BLENDING),
195                    write_mask: wgpu::ColorWrites::ALL,
196                })],
197                compilation_options: wgpu::PipelineCompilationOptions::default(),
198            }),
199            primitive: wgpu::PrimitiveState {
200                topology: wgpu::PrimitiveTopology::TriangleList,
201                ..Default::default()
202            },
203            depth_stencil: None,
204            multisample: wgpu::MultisampleState {
205                count: msaa_samples.max(1),
206                ..Default::default()
207            },
208            multiview: None,
209            cache: None,
210        });
211
212        let vertex_buffer = device.create_buffer(&wgpu::BufferDescriptor {
213            label: Some("gpu_graph_vertex"),
214            size: 4,
215            usage: wgpu::BufferUsages::VERTEX | wgpu::BufferUsages::COPY_DST,
216            mapped_at_creation: false,
217        });
218
219        Self {
220            pipeline,
221            bind_group,
222            uniform_buffer,
223            vertex_buffer,
224            vertex_capacity: 0,
225            target_format,
226            msaa_samples,
227        }
228    }
229}

231impl CallbackTrait for GraphCallback {
232    fn prepare(
233        &self,
234        device: &wgpu::Device,
235        queue: &wgpu::Queue,
236        screen_descriptor: &egui_wgpu::ScreenDescriptor,
237        _egui_encoder: &mut wgpu::CommandEncoder,
238        resources: &mut egui_wgpu::CallbackResources,
239    ) -> Vec<wgpu::CommandBuffer> {
240        let state = self.state.lock();
241        let Some(target_format) = state.target_format else {
242            return Vec::new();
243        };
244        if state.instances.is_empty() {
245            return Vec::new();
246        }
247
248        let needs_rebuild = match resources.get::<GraphRenderResources>() {
249            Some(existing) => {
250                existing.target_format != target_format
251                    || existing.msaa_samples != state.msaa_samples
252            }
253            None => true,
254        };
255
256        if needs_rebuild {
257            resources.insert(GraphRenderResources::new(
258                device,
259                target_format,
260                state.msaa_samples,
261            ));
262        }
263
264        let resources: &mut GraphRenderResources =
265            resources.get_mut::<GraphRenderResources>().unwrap();
266
267        let vertex_bytes = cast_slice(state.instances.as_slice());
268        let required_size = vertex_bytes.len();
269        if required_size > resources.vertex_capacity {
270            let new_size = required_size.next_power_of_two().max(256) as u64;
271            resources.vertex_buffer = device.create_buffer(&wgpu::BufferDescriptor {
272                label: Some("gpu_graph_vertex"),
273                size: new_size,
274                usage: wgpu::BufferUsages::VERTEX | wgpu::BufferUsages::COPY_DST,
275                mapped_at_creation: false,
276            });
277            resources.vertex_capacity = new_size as usize;
278        }
279        queue.write_buffer(&resources.vertex_buffer, 0, vertex_bytes);
280
281        let screen_size = [
282            screen_descriptor.size_in_pixels[0] as f32 / screen_descriptor.pixels_per_point,
283            screen_descriptor.size_in_pixels[1] as f32 / screen_descriptor.pixels_per_point,
284            0.0,
285            0.0,
286        ];
287        queue.write_buffer(&resources.uniform_buffer, 0, cast_slice(&screen_size));
288
289        Vec::new()
290    }
291
292    fn paint(
293        &self,
294        info: egui::PaintCallbackInfo,
295        render_pass: &mut wgpu::RenderPass<'static>,
296        resources: &egui_wgpu::CallbackResources,
297    ) {
298        let Some(resources) = resources.get::<GraphRenderResources>() else {
299            return;
300        };
301
302        let state = self.state.lock();
303        if state.instances.is_empty() {
304            return;
305        }
306
307        let clip = info.clip_rect_in_pixels();
308        render_pass.set_viewport(
309            0.0,
310            0.0,
311            info.screen_size_px[0] as f32,
312            info.screen_size_px[1] as f32,
313            0.0,
314            1.0,
315        );
316        render_pass.set_scissor_rect(
317            clip.left_px.max(0) as u32,
318            clip.top_px.max(0) as u32,
319            clip.width_px.max(0) as u32,
320            clip.height_px.max(0) as u32,
321        );
322        render_pass.set_pipeline(&resources.pipeline);
323        render_pass.set_bind_group(0, &resources.bind_group, &[]);
324        render_pass.set_vertex_buffer(0, resources.vertex_buffer.slice(..));
325        render_pass.draw(0..6, 0..(state.instances.len() as u32));
326    }
