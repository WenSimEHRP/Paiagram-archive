7impl ShapeInstance {
8    fn offset_perpendicular(a: Pos2, b: Pos2, offset_amount: f32) -> (Pos2, Pos2) {
9        let direction = b - a;
10        let direction_len = direction.length();
11        let unit_perpendicular = if direction_len > f32::EPSILON {
12            let unit_direction = direction / direction_len;
13            Vec2::new(-unit_direction.y, unit_direction.x)
14        } else {
15            Vec2::Y
16        };
17        let offset = unit_perpendicular * offset_amount;
18        (a + offset, b + offset)
19    }
20    pub fn segment(a: Pos2, b: Pos2, width: f32, color: Color32) -> Self {
21        let (a, b) = Self::offset_perpendicular(a, b, 1.5);
22        let rgba = color.to_array();
23        Self {
24            a: [a.x, a.y],
25            b: [b.x, b.y],
26            size: width,
27            color: [
28                rgba[0] as f32 / 255.0,
29                rgba[1] as f32 / 255.0,
30                rgba[2] as f32 / 255.0,
31                rgba[3] as f32 / 255.0,
32            ],
33            kind: 0,
34        }
35    }
36    pub fn circle(center: Pos2, radius: f32, color: Color32) -> Self {
37        let rgba = color.to_array();
38        Self {
39            a: [center.x, center.y],
40            b: [center.x, center.y],
41            size: radius,
42            color: [
43                rgba[0] as f32 / 255.0,
44                rgba[1] as f32 / 255.0,
45                rgba[2] as f32 / 255.0,
46                rgba[3] as f32 / 255.0,
47            ],
48            kind: 1,
49        }
50    }
51    pub fn stealth_arrow(from: Pos2, to: Pos2, center: Pos2, color: Color32) -> Self {
52        let direction = to - from;
53        let direction_len = direction.length();
54        let unit_direction = if direction_len > f32::EPSILON {
55            direction / direction_len
56        } else {
57            Vec2::X
58        };
59        let (a, b) = Self::offset_perpendicular(center, center + unit_direction, 1.5);
60        let rgba = color.to_array();
61        Self {
62            a: [a.x, a.y],
63            b: [b.x, b.y],
64            size: 14.0,
65            color: [
66                rgba[0] as f32 / 255.0,
67                rgba[1] as f32 / 255.0,
68                rgba[2] as f32 / 255.0,
69                rgba[3] as f32 / 255.0,
70            ],
71            kind: 2,
72        }
73    }

112    bind_group: wgpu::BindGroup,
113    uniform_buffer: wgpu::Buffer,
114    vertex_buffer: wgpu::Buffer,
115    vertex_capacity: usize,

120impl GraphRenderResources {
121    fn new(device: &wgpu::Device, target_format: wgpu::TextureFormat, msaa_samples: u32) -> Self {
122        let shader = device.create_shader_module(wgpu::ShaderModuleDescriptor {
123            label: Some("gpu_graph_shader"),
124            source: wgpu::ShaderSource::Wgsl(include_str!("gpu_graph.wgsl").into()),
125        });
126
127        let uniform_buffer = device.create_buffer(&wgpu::BufferDescriptor {
128            label: Some("gpu_graph_uniform"),
129            size: (std::mem::size_of::<[f32; 4]>() as u64).max(16),
130            usage: wgpu::BufferUsages::UNIFORM | wgpu::BufferUsages::COPY_DST,
131            mapped_at_creation: false,
132        });
133
134        let bind_group_layout = device.create_bind_group_layout(&wgpu::BindGroupLayoutDescriptor {
135            label: Some("gpu_graph_bind_group_layout"),
136            entries: &[wgpu::BindGroupLayoutEntry {
137                binding: 0,
138                visibility: wgpu::ShaderStages::VERTEX,
139                ty: wgpu::BindingType::Buffer {
140                    ty: wgpu::BufferBindingType::Uniform,
141                    has_dynamic_offset: false,
142                    min_binding_size: None,
143                },
144                count: None,
145            }],
146        });
147
148        let bind_group = device.create_bind_group(&wgpu::BindGroupDescriptor {
149            label: Some("gpu_graph_bind_group"),
150            layout: &bind_group_layout,
151            entries: &[wgpu::BindGroupEntry {
152                binding: 0,
153                resource: uniform_buffer.as_entire_binding(),
154            }],
155        });
156
157        let pipeline_layout = device.create_pipeline_layout(&wgpu::PipelineLayoutDescriptor {
158            label: Some("gpu_graph_pipeline_layout"),
159            bind_group_layouts: &[Some(&bind_group_layout)],
160            ..Default::default()
161        });
162
163        let pipeline = device.create_render_pipeline(&wgpu::RenderPipelineDescriptor {
164            label: Some("gpu_graph_pipeline"),
165            layout: Some(&pipeline_layout),
166            vertex: wgpu::VertexState {
167                module: &shader,
168                entry_point: Some("vs_main"),
169                buffers: &[wgpu::VertexBufferLayout {
170                    array_stride: std::mem::size_of::<ShapeInstance>() as u64,
171                    step_mode: wgpu::VertexStepMode::Instance,
172                    attributes: &[
173                        wgpu::VertexAttribute {
174                            format: wgpu::VertexFormat::Float32x2,
175                            offset: 0,
176                            shader_location: 0,
177                        },
178                        wgpu::VertexAttribute {
179                            format: wgpu::VertexFormat::Float32x2,
180                            offset: (2 * std::mem::size_of::<f32>()) as u64,
181                            shader_location: 1,
182                        },
183                        wgpu::VertexAttribute {
184                            format: wgpu::VertexFormat::Float32,
185                            offset: (4 * std::mem::size_of::<f32>()) as u64,
186                            shader_location: 2,
187                        },
188                        wgpu::VertexAttribute {
189                            format: wgpu::VertexFormat::Float32x4,
190                            offset: (5 * std::mem::size_of::<f32>()) as u64,
191                            shader_location: 3,
192                        },
193                        wgpu::VertexAttribute {
194                            format: wgpu::VertexFormat::Uint32,
195                            offset: (9 * std::mem::size_of::<f32>()) as u64,
196                            shader_location: 4,
197                        },
198                    ],
199                }],
200                compilation_options: wgpu::PipelineCompilationOptions::default(),
201            },
202            fragment: Some(wgpu::FragmentState {
203                module: &shader,
204                entry_point: Some("fs_main"),
205                targets: &[Some(wgpu::ColorTargetState {
206                    format: target_format,
207                    blend: Some(wgpu::BlendState::ALPHA_BLENDING),
208                    write_mask: wgpu::ColorWrites::ALL,
209                })],
210                compilation_options: wgpu::PipelineCompilationOptions::default(),
211            }),
212            primitive: wgpu::PrimitiveState {
213                topology: wgpu::PrimitiveTopology::TriangleList,
214                ..Default::default()
215            },
216            depth_stencil: None,
217            multisample: wgpu::MultisampleState {
218                count: msaa_samples.max(1),
219                ..Default::default()
220            },
221            multiview_mask: None,
222            cache: None,
223        });
224
225        let vertex_buffer = device.create_buffer(&wgpu::BufferDescriptor {
226            label: Some("gpu_graph_vertex"),
227            size: 4,
228            usage: wgpu::BufferUsages::VERTEX | wgpu::BufferUsages::COPY_DST,
229            mapped_at_creation: false,
230        });
231
232        Self {
233            pipeline,
234            bind_group,
235            uniform_buffer,
236            vertex_buffer,
237            vertex_capacity: 0,
238            target_format,
239            msaa_samples,
240        }
241    }
242}

244impl CallbackTrait for GraphCallback {
245    fn prepare(
246        &self,
247        device: &wgpu::Device,
248        queue: &wgpu::Queue,
249        screen_descriptor: &egui_wgpu::ScreenDescriptor,
250        _egui_encoder: &mut wgpu::CommandEncoder,
251        resources: &mut egui_wgpu::CallbackResources,
252    ) -> Vec<wgpu::CommandBuffer> {
253        let state = self.state.lock();
254        let Some(target_format) = state.target_format else {
255            return Vec::new();
256        };
257        if state.instances.is_empty() {
258            return Vec::new();
259        }
260
261        let needs_rebuild = match resources.get::<GraphRenderResources>() {
262            Some(existing) => {
263                existing.target_format != target_format
264                    || existing.msaa_samples != state.msaa_samples
265            }
266            None => true,
267        };
268
269        if needs_rebuild {
270            resources.insert(GraphRenderResources::new(
271                device,
272                target_format,
273                state.msaa_samples,
274            ));
275        }
276
277        let resources: &mut GraphRenderResources =
278            resources.get_mut::<GraphRenderResources>().unwrap();
279
280        let vertex_bytes = cast_slice(state.instances.as_slice());
281        let required_size = vertex_bytes.len();
282        if required_size > resources.vertex_capacity {
283            let new_size = required_size.next_power_of_two().max(256) as u64;
284            resources.vertex_buffer = device.create_buffer(&wgpu::BufferDescriptor {
285                label: Some("gpu_graph_vertex"),
286                size: new_size,
287                usage: wgpu::BufferUsages::VERTEX | wgpu::BufferUsages::COPY_DST,
288                mapped_at_creation: false,
289            });
290            resources.vertex_capacity = new_size as usize;
291        }
292        queue.write_buffer(&resources.vertex_buffer, 0, vertex_bytes);
293
294        let screen_size = [
295            screen_descriptor.size_in_pixels[0] as f32 / screen_descriptor.pixels_per_point,
296            screen_descriptor.size_in_pixels[1] as f32 / screen_descriptor.pixels_per_point,
297            0.0,
298            0.0,
299        ];
300        queue.write_buffer(&resources.uniform_buffer, 0, cast_slice(&screen_size));
301
302        Vec::new()
303    }
304
305    fn paint(
306        &self,
307        info: egui::PaintCallbackInfo,
308        render_pass: &mut wgpu::RenderPass<'static>,
309        resources: &egui_wgpu::CallbackResources,
310    ) {
311        let Some(resources) = resources.get::<GraphRenderResources>() else {
312            return;
313        };
314
315        let state = self.state.lock();
316        if state.instances.is_empty() {
317            return;
318        }
319
320        let clip = info.clip_rect_in_pixels();
321        render_pass.set_viewport(
322            0.0,
323            0.0,
324            info.screen_size_px[0] as f32,
325            info.screen_size_px[1] as f32,
326            0.0,
327            1.0,
328        );
329        render_pass.set_scissor_rect(
330            clip.left_px.max(0) as u32,
331            clip.top_px.max(0) as u32,
332            clip.width_px.max(0) as u32,
333            clip.height_px.max(0) as u32,
334        );
335        render_pass.set_pipeline(&resources.pipeline);
336        render_pass.set_bind_group(0, &resources.bind_group, &[]);
337        render_pass.set_vertex_buffer(0, resources.vertex_buffer.slice(..));
338        render_pass.draw(0..6, 0..(state.instances.len() as u32));
339    }
