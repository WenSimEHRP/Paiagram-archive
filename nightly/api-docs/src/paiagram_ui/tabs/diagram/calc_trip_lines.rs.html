46pub fn calc(
47    (InMut(buf), In(ctx)): (InMut<Vec<super::DrawnTrip>>, In<CalcContext>),
48    routes: Query<(&Route, &RouteTrips)>,
49    trip_q: Query<paiagram_core::trip::TripQuery>,
50    entries: Query<EntryQuery>,
51    stations: Query<(), With<Station>>,
52    platforms: Query<&ChildOf, With<Platform>>,
53    class_strokes: Query<&DisplayedStroke, With<Class>>,
54    settings: Res<ProjectSettings>,
55) {
56    buf.clear();
57
58    let Ok((route, trips)) = routes.get(ctx.route_entity) else {
59        return;
60    };
61
62    let heights: Vec<(Entity, f32)> = route.iter().collect();
63    let route_stops: EntityHashSet = route.stops.iter().copied().collect();
64    if heights.is_empty() {
65        return;
66    }
67
68    let vertical_visible = ctx.y_offset as f32
69        ..ctx.y_offset as f32 + ctx.screen_rect.height() / ctx.zoom_y.max(f32::EPSILON);
70
71    let visible_stations = {
72        let first_visible = heights
73            .iter()
74            .position(|(_, h)| *h > vertical_visible.start)
75            .or_else(|| {
76                heights
77                    .iter()
78                    .rposition(|(_, h)| *h <= vertical_visible.start)
79            });
80        let last_visible = heights.iter().rposition(|(_, h)| *h < vertical_visible.end);
81        if let (Some(first_visible), Some(mut last_visible)) = (first_visible, last_visible) {
82            let first_visible = first_visible.saturating_sub(2);
83            last_visible = (last_visible + 1).min(heights.len() - 1);
84            &heights[first_visible..=last_visible]
85        } else {
86            &[]
87        }
88    };
89
90    if visible_stations.is_empty() {
91        return;
92    }
93
94    let resolve_stop_station = |stop: Entity| -> Option<Entity> {
95        if route_stops.contains(&stop) {
96            return Some(stop);
97        }
98        if stations.get(stop).is_ok() {
99            return Some(stop);
100        }
101        let parent = platforms.get(stop).ok().map(|p| p.parent())?;
102        if route_stops.contains(&parent) {
103            return Some(parent);
104        }
105        None
106    };
107
108    #[derive(Clone, Copy)]
109    struct TripPoint {
110        arr: Pos2,
111        dep: Pos2,
112        entry: Entity,
113    }
114
115    #[derive(Clone, Copy)]
116    struct TripEntryData {
117        entity: Entity,
118        station: Entity,
119        arr_ticks: Option<Tick>,
120        dep_ticks: Option<Tick>,
121        has_departure: bool,
122    }
123
124    struct TripData {
125        entity: Entity,
126        stroke: DisplayedStroke,
127        entries: Vec<TripEntryData>,
128    }
129
130    let use_full_trip = true;
131    let stations_for_layout = if use_full_trip {
132        &heights[..]
133    } else {
134        visible_stations
135    };
136    let visible_station_set: EntityHashSet = visible_stations.iter().map(|(s, _)| *s).collect();
137    let mut station_index_map: EntityHashMap<Vec<usize>> = EntityHashMap::new();
138    for (idx, (station, _)) in stations_for_layout.iter().enumerate() {
139        station_index_map.entry(*station).or_default().push(idx);
140    }
141
142    let mut trip_data = Vec::with_capacity(trips.len());
143    for trip_entity in trips.iter().copied() {
144        let Ok(trip) = trip_q.get(trip_entity) else {
145            continue;
146        };
147
148        let stroke = class_strokes.get(trip.class.0).copied().unwrap_or_default();
149
150        let mut trip_entries_vec: Vec<TripEntryData> = Vec::new();
151        for entry_entity in trip.schedule.iter().copied() {
152            let Ok(entry) = entries.get(entry_entity) else {
153                continue;
154            };
155            let Some(station_entity) = resolve_stop_station(entry.stop()) else {
156                continue;
157            };
158            let (arr_ticks, dep_ticks) = if let Some(estimate) = entry.estimate {
159                let arrival_ticks = Tick::from_timetable_time(estimate.arr);
160                let departure_ticks = Tick::from_timetable_time(estimate.dep);
161                (Some(arrival_ticks), Some(departure_ticks))
162            } else {
163                (None, None)
164            };
165
166            trip_entries_vec.push(TripEntryData {
167                entity: entry_entity,
168                station: station_entity,
169                arr_ticks,
170                dep_ticks,
171                has_departure: entry.mode.arr.is_some(),
172            });
173        }
174
175        if trip_entries_vec.len() < 2 {
176            continue;
177        }
178
179        trip_data.push(TripData {
180            entity: trip_entity,
181            stroke,
182            entries: trip_entries_vec,
183        });
184    }
185
186    let repeat_freq_ticks = Tick::from_timetable_time(paiagram_core::units::time::TimetableTime(
187        settings.repeat_frequency.0,
188    ));
189
190    let drawn: Vec<super::DrawnTrip> = trip_data
191        .par_iter()
192        .filter_map(|trip| {
193            if use_full_trip
194                && !trip
195                    .entries
196                    .iter()
197                    .any(|entry| visible_station_set.contains(&entry.station))
198            {
199                return None;
200            }
201
202            let mut base_min: Option<Tick> = None;
203            let mut base_max: Option<Tick> = None;
204            for entry in &trip.entries {
205                let (Some(arrival_ticks), Some(departure_ticks)) =
206                    (entry.arr_ticks, entry.dep_ticks)
207                else {
208                    continue;
209                };
210                let local_min = Tick(arrival_ticks.0.min(departure_ticks.0));
211                let local_max = Tick(arrival_ticks.0.max(departure_ticks.0));
212                base_min = Some(base_min.map_or(local_min, |v| v.min(local_min)));
213                base_max = Some(base_max.map_or(local_max, |v| v.max(local_max)));
214            }
215
216            let Some(base_min) = base_min else {
217                return None;
218            };
219            let Some(base_max) = base_max else {
220                return None;
221            };
222
223            let (repeat_start, repeat_end) = if repeat_freq_ticks.0 > 0 {
224                let start =
225                    (ctx.visible_ticks.start.0 - base_max.0).div_euclid(repeat_freq_ticks.0);
226                let end = (ctx.visible_ticks.end.0 - base_min.0).div_euclid(repeat_freq_ticks.0);
227                (start, end)
228            } else {
229                (0, 0)
230            };
231
232            let mut drawn_segments = Vec::new();
233            let mut drawn_entries = Vec::new();
234
235            for repeat in repeat_start..=repeat_end {
236                let repeat_offset = repeat * repeat_freq_ticks.0;
237
238                let first_visible = trip.entries.iter().position(|entry| {
239                    let (Some(arrival_ticks), Some(departure_ticks)) =
240                        (entry.arr_ticks, entry.dep_ticks)
241                    else {
242                        return false;
243                    };
244                    let arrival_ticks = arrival_ticks.0 + repeat_offset;
245                    let departure_ticks = departure_ticks.0 + repeat_offset;
246                    !(departure_ticks < ctx.visible_ticks.start.0
247                        || arrival_ticks > ctx.visible_ticks.end.0)
248                });
249                let last_visible = trip.entries.iter().rposition(|entry| {
250                    let (Some(arrival_ticks), Some(departure_ticks)) =
251                        (entry.arr_ticks, entry.dep_ticks)
252                    else {
253                        return false;
254                    };
255                    let arrival_ticks = arrival_ticks.0 + repeat_offset;
256                    let departure_ticks = departure_ticks.0 + repeat_offset;
257                    !(departure_ticks < ctx.visible_ticks.start.0
258                        || arrival_ticks > ctx.visible_ticks.end.0)
259                });
260
261                let Some(first_visible) = first_visible else {
262                    continue;
263                };
264                let Some(last_visible) = last_visible else {
265                    continue;
266                };
267
268                let trip_entries = if use_full_trip {
269                    &trip.entries[..]
270                } else {
271                    let first_visible = first_visible.saturating_sub(2);
272                    let last_visible = (last_visible + 2).min(trip.entries.len() - 1);
273                    &trip.entries[first_visible..=last_visible]
274                };
275
276                if trip_entries.len() < 2 {
277                    continue;
278                }
279
280                let mut segments: Vec<Vec<TripPoint>> = Vec::new();
281
282                let mut local_edges: Vec<(Vec<TripPoint>, usize)> = Vec::new();
283                let mut previous_indices: Vec<usize> = Vec::new();
284
285                if let Some(first) = trip_entries.first() {
286                    if let Some(indices) = station_index_map.get(&first.station) {
287                        previous_indices = indices.clone();
288                    }
289                }
290
291                for entry_idx in 0..trip_entries.len() {
292                    let entry = &trip_entries[entry_idx];
293                    let next = trip_entries.get(entry_idx + 1);
294
295                    if previous_indices.is_empty() {
296                        if let Some(next_entry) = next {
297                            if let Some(indices) = station_index_map.get(&next_entry.station) {
298                                previous_indices = indices.clone();
299                            }
300                        }
301                        for (segment, _) in local_edges.drain(..) {
302                            if segment.len() >= 2 {
303                                segments.push(segment);
304                            }
305                        }
306                        continue;
307                    }
308
309                    let (Some(arrival_ticks), Some(departure_ticks)) =
310                        (entry.arr_ticks, entry.dep_ticks)
311                    else {
312                        for (segment, _) in local_edges.drain(..) {
313                            if segment.len() >= 2 {
314                                segments.push(segment);
315                            }
316                        }
317                        if let Some(next_entry) = next {
318                            if let Some(indices) = station_index_map.get(&next_entry.station) {
319                                previous_indices = indices.clone();
320                            }
321                        }
322                        continue;
323                    };
324
325                    let arrival_ticks = Tick(arrival_ticks.0 + repeat_offset);
326                    let departure_ticks = Tick(departure_ticks.0 + repeat_offset);
327
328                    let mut next_local_edges = Vec::new();
329
330                    for &current_line_index in &previous_indices {
331                        let Some((_, height)) = stations_for_layout.get(current_line_index) else {
332                            continue;
333                        };
334
335                        let matched_idx = local_edges
336                            .iter()
337                            .position(|(_, idx)| current_line_index.abs_diff(*idx) <= 1);
338
339                        let mut segment = if let Some(idx) = matched_idx {
340                            local_edges.swap_remove(idx).0
341                        } else {
342                            Vec::new()
343                        };
344
345                        let arrival_pos = Pos2::new(
346                            super::draw_lines::ticks_to_screen_x(
347                                arrival_ticks.0,
348                                ctx.screen_rect,
349                                ctx.ticks_per_screen_unit,
350                                ctx.x_offset.0,
351                            ),
352                            (height - ctx.y_offset as f32) * ctx.zoom_y + ctx.screen_rect.top(),
353                        );
354
355                        let departure_pos = Pos2::new(
356                            super::draw_lines::ticks_to_screen_x(
357                                departure_ticks.0,
358                                ctx.screen_rect,
359                                ctx.ticks_per_screen_unit,
360                                ctx.x_offset.0,
361                            ),
362                            (height - ctx.y_offset as f32) * ctx.zoom_y + ctx.screen_rect.top(),
363                        );
364
365                        segment.push(TripPoint {
366                            arr: arrival_pos,
367                            dep: departure_pos,
368                            entry: entry.entity,
369                        });
370
371                        let mut continued = false;
372                        if let Some(next_entry) = next {
373                            for offset in [-1, 0, 1] {
374                                let next_idx = (current_line_index as isize + offset) as usize;
375                                if let Some((s, _)) = stations_for_layout.get(next_idx) {
376                                    if *s == next_entry.station {
377                                        next_local_edges.push((segment.clone(), next_idx));
378                                        continued = true;
379                                        break;
380                                    }
381                                }
382                            }
383                        }
384
385                        if !continued && segment.len() >= 2 {
386                            segments.push(segment);
387                        }
388                    }
389
390                    for (segment, _) in local_edges.drain(..) {
391                        if segment.len() >= 2 {
392                            segments.push(segment);
393                        }
394                    }
395
396                    local_edges = next_local_edges;
397                    if let Some(next_entry) = next {
398                        if let Some(indices) = station_index_map.get(&next_entry.station) {
399                            previous_indices = indices.clone();
400                        }
401                    }
402                }
403
404                for (segment, _) in local_edges {
405                    if segment.len() >= 2 {
406                        segments.push(segment);
407                    }
408                }
409
410                if segments.is_empty() {
411                    continue;
412                }
413
414                for segment in segments {
415                    let mut cubics = Vec::new();
416                    let mut segment_entries = Vec::new();
417
418                    for point in segment {
419                        cubics.push([point.arr, point.arr, point.dep, point.dep]);
420                        segment_entries.push(point.entry);
421                    }
422
423                    if !cubics.is_empty() {
424                        drawn_segments.push(cubics);
425                        drawn_entries.push(segment_entries);
426                    }
427                }
428            }
429
430            if drawn_segments.is_empty() {
431                return None;
432            }
433
434            Some(super::DrawnTrip {
435                entity: trip.entity,
436                stroke: trip.stroke,
437                points: drawn_segments,
438                entries: drawn_entries,
439            })
440        })
441        .collect();
442
443    buf.extend(drawn);
444}
