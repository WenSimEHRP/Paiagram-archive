163pub fn rewrite_trip_cache(
164    cache: &super::TripCache,
165    stations: impl Iterator<Item = f32>,
166    state: &mut GpuTripRendererState,
167) {
168    state.trips.clear();
169    state.entries.clear();
170    state.stations.clear();
171    state.stations.extend(stations);
172    for (_trip_entity, lines) in cache.iter() {
173        for line in lines {
174            state.trips.push(Trip {
175                color: Color32::MAGENTA.to_array(),
176                width: 1.0,
177                len: line.len() as u32,
178                start_idx: state.entries.len() as u32,
179            });
180            for entry in line {
181                state.entries.push(Entry {
182                    dep_secs: entry.dep.seconds(),
183                    arr_secs: entry.arr.seconds(),
184                    station_index: entry.station_index as u32,
185                    _track_index: 0,
186                });
187            }
188        }
189    }
190}

526impl CallbackTrait for TripCallback {
527    fn prepare(
528        &self,
529        device: &wgpu::Device,
530        queue: &wgpu::Queue,
531        screen_descriptor: &egui_wgpu::ScreenDescriptor,
532        egui_encoder: &mut wgpu::CommandEncoder,
533        resources: &mut egui_wgpu::CallbackResources,
534    ) -> Vec<wgpu::CommandBuffer> {
535        let state = self.state.lock();
536        let Some(target_format) = state.target_format else {
537            return Vec::new();
538        };
539
540        let state_key = Arc::as_ptr(&self.state) as usize;
541
542        if resources.get::<TripRenderResourceMap>().is_none() {
543            resources.insert(TripRenderResourceMap::default());
544        }
545
546        let resources_map: &mut TripRenderResourceMap =
547            resources.get_mut::<TripRenderResourceMap>().unwrap();
548
549        let needs_rebuild = match resources_map.by_state.get(&state_key) {
550            Some(existing) => {
551                existing.target_format != target_format
552                    || existing.msaa_samples != state.msaa_samples
553            }
554            None => true,
555        };
556
557        if needs_rebuild {
558            resources_map.by_state.insert(
559                state_key,
560                TripRenderResources::new(device, target_format, state.msaa_samples),
561            );
562        }
563
564        let resources: &mut TripRenderResources =
565            resources_map.by_state.get_mut(&state_key).unwrap();
566
567        let mut needs_rebind = false;
568
569        let mut data_tick_min = i32::MAX;
570        let mut data_tick_max = i32::MIN;
571        for entry in &state.entries {
572            let arr_ticks = entry.arr_secs.saturating_mul(100);
573            let dep_ticks = entry.dep_secs.saturating_mul(100);
574            data_tick_min = data_tick_min.min(arr_ticks.min(dep_ticks));
575            data_tick_max = data_tick_max.max(arr_ticks.max(dep_ticks));
576        }
577        if data_tick_min > data_tick_max {
578            data_tick_min = 0;
579            data_tick_max = 0;
580        }
581
582        // Entries
583        {
584            let entry_bytes = cast_slice(state.entries.as_slice());
585            let required_size = entry_bytes.len();
586            if required_size as u64 > resources.entry_buffer.size() {
587                let new_size = required_size.next_power_of_two().max(256) as u64;
588                resources.entry_buffer = make_storage_buffer_entry("entries", new_size, device);
589                needs_rebind = true;
590            }
591            queue.write_buffer(&resources.entry_buffer, 0, entry_bytes);
592        }
593
594        // Trips
595        {
596            let trip_bytes = cast_slice(state.trips.as_slice());
597            let required_size = trip_bytes.len();
598            if required_size as u64 > resources.trip_buffer.size() {
599                let new_size = required_size.next_power_of_two().max(256) as u64;
600                resources.trip_buffer = make_storage_buffer_entry("trips", new_size, device);
601                needs_rebind = true;
602            }
603            queue.write_buffer(&resources.trip_buffer, 0, trip_bytes);
604        }
605
606        // Stations
607        {
608            let station_bytes = cast_slice(state.stations.as_slice());
609            let required_size = station_bytes.len();
610            if required_size as u64 > resources.station_buffer.size() {
611                let new_size = required_size.next_power_of_two().max(256) as u64;
612                resources.station_buffer = make_storage_buffer_entry("stations", new_size, device);
613                needs_rebind = true;
614            }
615            queue.write_buffer(&resources.station_buffer, 0, station_bytes);
616        }
617
618        // Direct index map source: instance_index -> (trip_index, local_segment)
619        let mut instance_map: Vec<InstanceMapEntry> = Vec::new();
620        for (trip_index, trip) in state.trips.iter().enumerate() {
621            let seg_count = trip.len.saturating_sub(1);
622            for local_segment in 0..seg_count {
623                instance_map.push(InstanceMapEntry {
624                    trip_index: trip_index as u32,
625                    local_segment,
626                });
627            }
628        }
629        let map_bytes = cast_slice(instance_map.as_slice());
630        let required_size = map_bytes.len();
631        if required_size as u64 > resources.source_instance_map_buffer.size() {
632            let new_size = required_size.next_power_of_two().max(256) as u64;
633            resources.source_instance_map_buffer =
634                make_storage_buffer_entry("source_instance_map", new_size, device);
635            needs_rebind = true;
636        }
637        queue.write_buffer(&resources.source_instance_map_buffer, 0, map_bytes);
638
639        if needs_rebind {
640            resources.rebuild_bind_group(device);
641        }
642
643        // uniforms
644        let uniforms = GpuUniforms {
645            ticks_min: state
646                .uniforms
647                .ticks_min
648                .clamp(i32::MIN as i64, i32::MAX as i64) as i32,
649            y_min: state.uniforms.y_min as f32,
650            screen_size: [
651                screen_descriptor.size_in_pixels[0] as f32 / screen_descriptor.pixels_per_point,
652                screen_descriptor.size_in_pixels[1] as f32 / screen_descriptor.pixels_per_point,
653            ],
654            x_per_unit: state.uniforms.x_per_unit,
655            y_per_unit: state.uniforms.y_per_unit,
656            screen_origin: state.uniforms.screen_origin,
657            ..Default::default()
658        };
659        let visible_ticks_min = uniforms.ticks_min;
660        let visible_ticks_max = uniforms
661            .ticks_min
662            .saturating_add((uniforms.screen_size[0] * uniforms.x_per_unit) as i32);
663        let repeat_interval = state.uniforms.repeat_interval_ticks.max(0);
664        let (repeat_from, repeat_to) = if repeat_interval > 0 {
665            (
666                (visible_ticks_min - data_tick_max).div_euclid(repeat_interval),
667                (visible_ticks_max - data_tick_min).div_euclid(repeat_interval),
668            )
669        } else {
670            (0, 0)
671        };
672        let repeat_count = if repeat_interval > 0 {
673            (repeat_to - repeat_from + 1).max(1) as usize
674        } else {
675            1usize
676        };
677        let visible_capacity = instance_map.len().saturating_mul(repeat_count);
678        resources.draw_instance_count = visible_capacity.min(u32::MAX as usize) as u32;
679        let visible_required_size = visible_capacity
680            .saturating_mul(std::mem::size_of::<VisibleSegment>())
681            .max(256) as u64;
682        if visible_required_size > resources.visible_segment_buffer.size() {
683            let new_size = visible_required_size.next_power_of_two().max(256);
684            resources.visible_segment_buffer = make_rw_storage_buffer_entry(
685                "visible_segment",
686                new_size,
687                BufferUsages::VERTEX,
688                device,
689            );
690            resources.visible_segment_write_buffer = make_rw_storage_buffer_entry(
691                "visible_segment_write",
692                new_size,
693                BufferUsages::COPY_SRC,
694                device,
695            );
696            needs_rebind = true;
697        }
698        if needs_rebind {
699            resources.rebuild_bind_group(device);
700        }
701
702        let uniforms = GpuUniforms {
703            repeat_interval_ticks: repeat_interval,
704            repeat_from,
705            repeat_to,
706            source_instance_count: (instance_map.len() as u32).min(u32::MAX),
707            feathering_radius: 1.2 / screen_descriptor.pixels_per_point,
708            ..uniforms
709        };
710        let uniform_bytes = bytes_of(&uniforms);
711        queue.write_buffer(&resources.uniform_buffer, 0, uniform_bytes);
712
713        // With direct draw we render a fixed slot count, so clear unused slots each frame.
714        egui_encoder.clear_buffer(&resources.visible_segment_write_buffer, 0, None);
715
716        {
717            let mut pass = egui_encoder.begin_compute_pass(&wgpu::ComputePassDescriptor {
718                label: Some("gpu_trip_cull_compute_pass"),
719                timestamp_writes: None,
720            });
721            pass.set_pipeline(&resources.compute_pipeline);
722            pass.set_bind_group(0, &resources.bind_group, &[]);
723            let workgroup_size = 64u32;
724            let workgroup_count =
725                ((instance_map.len() as u32).saturating_add(workgroup_size - 1)) / workgroup_size;
726            if workgroup_count > 0 {
727                pass.dispatch_workgroups(workgroup_count, 1, 1);
728            }
729        }
730        egui_encoder.copy_buffer_to_buffer(
731            &resources.visible_segment_write_buffer,
732            0,
733            &resources.visible_segment_buffer,
734            0,
735            visible_required_size,
736        );
737
738        Vec::new()
739    }
740
741    fn paint(
742        &self,
743        info: egui::PaintCallbackInfo,
744        render_pass: &mut wgpu::RenderPass<'static>,
745        resources: &egui_wgpu::CallbackResources,
746    ) {
747        let Some(resources_map) = resources.get::<TripRenderResourceMap>() else {
748            return;
749        };
750        let state_key = Arc::as_ptr(&self.state) as usize;
751        let Some(resources) = resources_map.by_state.get(&state_key) else {
752            return;
753        };
754
755        let clip = info.clip_rect_in_pixels();
756        let clip_left = clip.left_px.max(0) as u32;
757        let clip_top = clip.top_px.max(0) as u32;
758        let clip_width = clip.width_px.max(0) as u32;
759        let clip_height = clip.height_px.max(0) as u32;
760        render_pass.set_viewport(
761            0.0,
762            0.0,
763            info.screen_size_px[0] as f32,
764            info.screen_size_px[1] as f32,
765            0.0,
766            1.0,
767        );
768        render_pass.set_scissor_rect(clip_left, clip_top, clip_width, clip_height);
769        render_pass.set_pipeline(&resources.pipeline);
770        render_pass.set_bind_group(0, &resources.bind_group, &[]);
771        render_pass.set_vertex_buffer(0, resources.visible_segment_buffer.slice(..));
772        render_pass.set_vertex_buffer(1, resources.segment_mesh_vertex_buffer.slice(..));
773        render_pass.set_index_buffer(
774            resources.segment_mesh_index_buffer.slice(..),
775            wgpu::IndexFormat::Uint16,
776        );
777        render_pass.draw_indexed(
778            0..SEGMENT_MESH_INDICES.len() as u32,
779            0,
780            0..resources.draw_instance_count,
781        );
782    }
