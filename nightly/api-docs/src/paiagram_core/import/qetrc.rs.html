20/// The root structure of the qETRC JSON data
21#[derive(Deserialize)]
22struct Root {
23    // qetrc_release: u32,
24    // qetrc_version: String,
25    /// Trains in the original qETRC data. Each "train" corresponds to a [`crate::trip::Trip`] in Paiagram.
26    #[serde(rename = "trains")]
27    services: Vec<Service>,
28    // qETRC has the line field and the lines array, both contains line data.
29    // pyETRC only has the `line` field, while qETRC uses both to support multiple lines.
30    // To keep compatibility with pyETRC, we keep the `line` field as is,
31    // The lines would be chained together later with std::iter::once and chain
32    /// A single [`Line`]
33    line: Line,
34    /// Additional [`Line`]s. This field does not exist in pyETRC, only in qETRC.
35    lines: Option<Vec<Line>>,
36    /// Vehicles in the qETRC data.
37    /// They are named "circuits" in the original qETRC data. A "circuit" refers to a train that runs a set of services
38    /// in a given period, which matches the concept of [`Vehicle`] in Paiagram.
39    #[serde(rename = "circuits")]
40    vehicles: Vec<Vehicle>,
41    config: Option<Config>,
42}

44/// A line that is used as the foundation of connection in qETRC data
45#[derive(Deserialize)]
46struct Line {
47    /// The name of the line
48    name: String,
49    /// [`Station`]s on the line.
50    stations: Vec<QStation>,

53#[derive(Deserialize)]
54struct QStation {
55    /// Station name
56    #[serde(rename = "zhanming")]
57    name: String,
58    /// Distance from the start of the line, in kilometers
59    #[serde(rename = "licheng")]
60    distance_km: f32,

63#[derive(Deserialize)]
64struct Service {
65    /// Each service may have multiple service numbers.
66    /// In qETRC's case, the first service number is always the main one, and we use that one in Paiagram.
67    #[serde(rename = "checi")]
68    service_number: Vec<String>,
69    #[serde(rename = "type")]
70    service_type: String,
71    /// The timetable entries of the service
72    timetable: Vec<TimetableEntry>,
73}

75#[derive(Deserialize)]
76struct TimetableEntry {
77    /// Whether the train would stop and load/unload passengers or freight at the station.
78    #[serde(rename = "business")]
79    would_stop: Option<bool>,
80    /// Arrival time in "HH:MM" format. "ddsj" in the original qETRC data refers to "到达时间".
81    #[serde(rename = "ddsj")]
82    arrival: String,
83    /// Departure time in "HH:MM" format. "cfsj" in the original qETRC data refers to "出发时间".
84    #[serde(rename = "cfsj")]
85    departure: String,
86    /// Station name
87    #[serde(rename = "zhanming")]
88    station_name: String,
89}

116pub fn load_qetrc(event: On<super::LoadQETRC>, mut commands: Commands, mut graph: ResMut<Graph>) {
117    let root: Root = match serde_json::from_str(&event.content) {
118        Ok(r) => r,
119        // TODO: handle warning better
120        // TODO: add log page and warning banner
121        Err(e) => {
122            warn!("Failed to parse QETRC data: {e:?}");
123            return;
124        }
125    };
126    let lines_iter = std::iter::once(root.line).chain(root.lines.into_iter().flatten());
127    let mut station_map: HashMap<String, Instance<Station>> = HashMap::new();
128    let mut class_map: HashMap<String, Instance<Class>> = HashMap::new();
129    if let Some(config) = root.config {
130        for (class, color) in config.default_colors {
131            // #RRGGBB
132            // 0123456
133            let (r, g, b) = (
134                u8::from_str_radix(&color[1..=2], 16).unwrap(),
135                u8::from_str_radix(&color[3..=4], 16).unwrap(),
136                u8::from_str_radix(&color[5..=6], 16).unwrap(),
137            );
138            super::make_class(&class, &mut class_map, &mut commands, || ClassBundle {
139                class: Class::default(),
140                name: Name::new(class.clone()),
141                stroke: DisplayedStroke {
142                    width: 1.0,
143                    color: DisplayedColor::Custom(Color32::from_rgb(r, g, b)),
144                },
145            });
146        }
147    }
148    for line in lines_iter {
149        let mut entity_distances: Vec<(Instance<Station>, f32)> =
150            Vec::with_capacity(line.stations.len());
151        for station in line.stations {
152            let e = super::make_station(&station.name, &mut station_map, &mut graph, &mut commands);
153            entity_distances.push((e, station.distance_km));
154        }
155        for w in entity_distances.windows(2) {
156            let [(prev, prev_d), (this, this_d)] = w else {
157                unreachable!()
158            };
159            // TODO: handle one way stations and intervals
160            let length = Distance::from_km((this_d - prev_d).abs());
161            super::add_interval_pair(
162                &mut graph,
163                &mut commands,
164                prev.entity(),
165                this.entity(),
166                length,
167            );
168        }
169        let mut previous_distance_km = entity_distances.first().map_or(0.0, |(_, d)| *d);
170        for (_, distance_km) in entity_distances.iter_mut().skip(1) {
171            let current_distance_km = *distance_km;
172            *distance_km -= previous_distance_km;
173            previous_distance_km = current_distance_km;
174        }
175        // create a new displayed line
176        commands.spawn((
177            Name::new(line.name),
178            Route {
179                stops: entity_distances.iter().map(|(e, _)| e.entity()).collect(),
180                lengths: entity_distances.iter().copied().map(|(_, d)| d).collect(),
181            },
182        ));
183    }
184    let mut trip_pool: HashMap<String, Entity> = HashMap::with_capacity(root.services.len());
185    for service in root.services {
186        let mut entries: Vec<_> = service
187            .timetable
188            .into_iter()
189            .map(|e| {
190                (
191                    TimetableTime::from_str(&e.arrival).unwrap(),
192                    TimetableTime::from_str(&e.departure).unwrap(),
193                    super::make_station(
194                        &e.station_name,
195                        &mut station_map,
196                        &mut graph,
197                        &mut commands,
198                    ),
199                )
200            })
201            .collect();
202        super::normalize_times(entries.iter_mut().flat_map(|(a, d, _)| [a, d]));
203        let trip_class =
204            super::make_class(&service.service_type, &mut class_map, &mut commands, || {
205                ClassBundle {
206                    class: Class::default(),
207                    name: Name::new(service.service_type.clone()),
208                    stroke: DisplayedStroke::from_seed(service.service_type.as_bytes()),
209                }
210            });
211        let nominal_entries: Vec<Entity> = entries
212            .into_iter()
213            .map(|(arr, dep, stop)| {
214                if dep < arr {
215                    info!(?arr, ?dep, ?service.service_number)
216                }
217                debug_assert!(dep >= arr);
218                let arr = (dep != arr).then(|| TravelMode::At(arr));
219                let dep = TravelMode::At(dep);
220                commands
221                    .spawn(EntryBundle::new(arr, dep, stop.entity()))
222                    .id()
223            })
224            .collect();
225        let trip_entity = commands
226            .spawn_empty()
227            .add_children(&nominal_entries)
228            .insert(TripBundle::new(
229                &service.service_number[0],
230                TripClass(trip_class.entity()),
231                nominal_entries,
232            ))
233            .id();
234        trip_pool.insert(service.service_number[0].clone(), trip_entity);
235    }
236    for vehicle in root.vehicles {
237        let vehicle_name = format!("{} [{}]", vehicle.name, vehicle.make);
238        let mut v = crate::vehicle::Vehicle::default();
239        for number in vehicle.services.iter().map(|it| &it.service_number) {
240            let Some(&e) = trip_pool.get(number) else {
241                warn!(
242                    "Vehicle {} has trip {} but the trip isn't in pool",
243                    vehicle_name, number
244                );
245                continue;
246            };
247            v.trips.push(e);
248        }
249        commands.spawn((Name::new(vehicle_name), v));
250    }
251}
