68impl MapEntities for Graph {
69    fn map_entities<E: EntityMapper>(&mut self, entity_mapper: &mut E) {
70        // construct a new graph instead
71        let (nodes, edges) = self.capacity();
72        let mut new_graph = DiGraphMap::with_capacity(nodes, edges);
73        for mut node in self.nodes() {
74            node.map_entities(entity_mapper);
75            new_graph.add_node(node);
76        }
77        for (mut source, mut target, weight) in self.all_edges() {
78            let mut weight = *weight;
79            source.map_entities(entity_mapper);
80            target.map_entities(entity_mapper);
81            weight.map_entities(entity_mapper);
82            new_graph.add_edge(source, target, weight);
83        }
84        self.map = new_graph;
85    }
86}

88impl Graph {
89    pub fn route_between<'w>(
90        &self,
91        source: Entity,
92        target: Entity,
93        interval_q: &Query<'w, 'w, IntervalQuery>,
94    ) -> Option<(i32, Vec<Entity>)> {
95        astar(
96            &self.map,
97            source,
98            |f| f == target,
99            |e| {
100                let Ok(i) = interval_q.get(*e.weight()) else {
101                    return i32::MAX;
102                };
103                i.distance().0
104            },
105            |_| 0,
106        )
107    }
108    pub fn into_graph(self) -> petgraph::Graph<Entity, Entity> {
109        self.map.into_graph()
110    }

136        let dx = self.point[0] - point[0];

303const EARTH_RADIUS_METERS: f64 = 6_378_137.0;
304const WEB_MERCATOR_MAX_LAT: f64 = 85.051_128_78;
305
306pub fn lon_lat_to_xy(lon: f64, lat: f64) -> (f64, f64) {
307    let x = EARTH_RADIUS_METERS * lon.to_radians();
308    let lat = lat.clamp(-WEB_MERCATOR_MAX_LAT, WEB_MERCATOR_MAX_LAT);
309    let lat_rad = lat.to_radians();
310    let y = -EARTH_RADIUS_METERS * (std::f64::consts::FRAC_PI_4 + lat_rad / 2.0).tan().ln();
311    (x, y)
312}
313
314pub fn xy_to_lon_lat(x: f64, y: f64) -> (f64, f64) {
315    let lon = (x / EARTH_RADIUS_METERS).to_degrees();
316    let lat = (2.0 * (-y / EARTH_RADIUS_METERS).exp().atan() - std::f64::consts::FRAC_PI_2)
317        .to_degrees();
318    (lon, lat)
319}

351fn start_graph_spatial_index_rebuild(
352    mut state: ResMut<GraphSpatialIndexState>,
353    nodes: Query<(Entity, &Node)>,
354) {
355    if !state.dirty || state.task.is_some() {
356        return;
357    }
358    state.dirty = false;
359
360    let snapshot: Vec<(Entity, [f64; 2])> = nodes
361        .iter()
362        .map(|(entity, node)| (entity, node.pos.to_xy_arr()))
363        .collect();
364    state.task = Some(AsyncComputeTaskPool::get().spawn(async move {
365        let entries: Vec<SpatialIndexedEntity> = snapshot
366            .into_iter()
367            .map(|(entity, point)| SpatialIndexedEntity { entity, point })
368            .collect();
369        RTree::bulk_load(entries)
370    }));
371}

373fn start_graph_interval_spatial_index_rebuild(
374    mut state: ResMut<GraphIntervalSpatialIndexState>,
375    graph: Res<Graph>,
376    nodes: Query<&Node>,
377) {
378    if !state.dirty || state.task.is_some() {
379        return;
380    }
381    state.dirty = false;
382
383    let mut snapshot = Vec::<IntervalSpatialIndexedEntity>::new();
384    for (source, target, interval) in graph.all_edges() {
385        let Ok(source_node) = nodes.get(source) else {
386            continue;
387        };
388        let Ok(target_node) = nodes.get(target) else {
389            continue;
390        };
391        snapshot.push(IntervalSpatialIndexedEntity {
392            interval: *interval,
393            p0: source_node.pos.to_xy_arr(),
394            p1: target_node.pos.to_xy_arr(),
395        });
396    }
397
398    state.task = Some(AsyncComputeTaskPool::get().spawn(async move { RTree::bulk_load(snapshot) }));
399}

429/// The position of the node.
430///
431/// This stores longitude and latitude values only.
432#[derive(Reflect, Clone, Copy, Debug)]
433pub struct NodePos {
434    pub lon: f64,
435    pub lat: f64,
436}
437
438impl Default for NodePos {
439    fn default() -> Self {
440        Self::new(0.0, 0.0)
441    }
442}
443
444impl NodePos {
445    pub fn new(lon: f64, lat: f64) -> Self {
446        Self { lon, lat }
447    }
448    pub fn from_xy(x: f64, y: f64) -> Self {
449        let (lon, lat) = xy_to_lon_lat(x, y);
450        Self::new(lon, lat)
451    }
452    pub fn to_xy(&self) -> (f64, f64) {
453        lon_lat_to_xy(self.lon, self.lat)
454    }
455    pub fn to_xy_arr(&self) -> [f64; 2] {
456        let (x, y) = self.to_xy();
457        [x, y]
458    }
459    /// Shift the node on the canvas by x and y
460    pub fn shift(&mut self, dx: f64, dy: f64) {
461        self.lon += dx;
462        self.lat += dy;
463    }
464    /// Linearly interpolates between `self` and `other` by fraction `t`.
465    /// `t` is typically between 0.0 and 1.0.
466    pub fn lerp(&self, other: &Self, t: f64) -> Self {
467        let end_lon = other.lon;
468        let end_lat = other.lat;
469
470        Self {
471            lon: self.lon + (end_lon - self.lon) * t,
472            lat: self.lat + (end_lat - self.lat) * t,
473        }
474    }
475}

483fn update_graph_on_station_removal(
484    removed_station: On<Remove, Station>,
485    mut commands: Commands,
486    mut graph: ResMut<Graph>,
487) {
488    let s = removed_station.entity;
489    for e in graph
490        .neighbors_directed(s, petgraph::Direction::Incoming)
491        .chain(graph.neighbors_directed(s, petgraph::Direction::Outgoing))
492    {
493        commands.entity(e).despawn();
494    }
495    graph.remove_node(s);
496}
497
498fn update_graph_on_interval_removal(
499    removed_interval: On<Remove, Interval>,
500    mut graph: ResMut<Graph>,
501) {
502    let i = removed_interval.entity;
503    let mut source = None;
504    let mut target = None;
505    for (s, t, weight) in graph.all_edges() {
506        if i != *weight {
507            continue;
508        }
509        source = Some(s);
510        target = Some(t);
511        break;
512    }
513    let (Some(s), Some(t)) = (source, target) else {
514        return;
515    };
516    graph.remove_edge(s, t);
517}

519#[cfg(debug_assertions)]
520fn check_stations_in_graph(
521    graph: Res<Graph>,
522    stations: Populated<Entity, With<Station>>,
523    intervals: Populated<Entity, With<Interval>>,
524    names: Query<&Name>,
525) {
526    let queried_station_set: EntityHashSet = stations.iter().collect();
527    let queried_interval_set: EntityHashSet = intervals.iter().collect();
528    let mut graphed_station_set = EntityHashSet::new();
529    let mut graphed_interval_set = EntityHashSet::new();
530    for (_, _, w) in graph.all_edges() {
531        graphed_interval_set.insert(*w);
532    }
533    for node in graph.nodes() {
534        graphed_station_set.insert(node);
535    }
536    if queried_station_set != graphed_station_set {
537        debug_graph_set_diff(
538            "station",
539            &queried_station_set,
540            &graphed_station_set,
541            &names,
542        );
543    }
544    if queried_interval_set != graphed_interval_set {
545        debug_graph_set_diff(
546            "interval",
547            &queried_interval_set,
548            &graphed_interval_set,
549            &names,
550        );
551    }
552    debug_assert_eq!(queried_station_set, graphed_station_set);
553    debug_assert_eq!(queried_interval_set, graphed_interval_set);
554}

556#[cfg(debug_assertions)]
557fn debug_graph_set_diff(
558    label: &str,
559    queried: &EntityHashSet,
560    graphed: &EntityHashSet,
561    names: &Query<&Name>,
562) {
563    let intersection: EntityHashSet = queried.intersection(graphed).copied().collect();
564    let only_queried: EntityHashSet = queried.difference(graphed).copied().collect();
565    let only_graphed: EntityHashSet = graphed.difference(queried).copied().collect();
566
567    let list_with_names = |set: &EntityHashSet| -> Vec<String> {
568        let mut out: Vec<String> = set
569            .iter()
570            .map(|e| match names.get(*e) {
571                Ok(name) => format!("{} ({})", name.as_str(), e.index()),
572                Err(_) => format!("<unnamed> ({})", e.index()),
573            })
574            .collect();
575        out.sort_unstable();
576        out
577    };
578
579    warn!(
580        "Graph {label} set mismatch: intersection={:#?} | only_queried={:#?} | only_graphed={:#?}",
581        list_with_names(&intersection),
582        list_with_names(&only_queried),
583        list_with_names(&only_graphed)
584    );
585}

587// TODO: instead of merging them, make stations platforms instead
588pub fn merge_station_by_name(
589    mut commands: Commands,
590    mut graph: ResMut<Graph>,
591    stations: Query<(Entity, &Name, &Platforms), With<Station>>,
592    entry_stops: Query<(Entity, &EntryStop)>,
593    mut routes: Query<&mut Route>,
594) {
595    let mut name_map: HashMap<&str, SmallVec<[Entity; 1]>> = HashMap::new();
596    for (entity, name, _) in &stations {
597        let v = name_map.entry(name.as_str()).or_default();
598        v.push(entity);
599    }
600
601    let mut remap: EntityHashMap<Entity> = EntityHashMap::default();
602
603    for (_name, mut entities) in name_map.into_iter().filter(|(_, v)| v.len() > 1) {
604        entities.sort_unstable_by_key(|entity| entity.index());
605        let keep = entities[0];
606
607        for duplicate in entities.into_iter().skip(1) {
608            if let Ok((_, _, platforms)) = stations.get(duplicate) {
609                let to_move: SmallVec<[Entity; 8]> = platforms.iter().collect();
610                if !to_move.is_empty() {
611                    commands.entity(keep).add_children(&to_move);
612                }
613            }
614            remap.insert(duplicate, keep);
615        }
616    }
617
618    if remap.is_empty() {
619        return;
620    }
621
622    let (nodes, edges) = graph.capacity();
623    let mut new_graph = DiGraphMap::with_capacity(nodes, edges);
624    let mut removed_intervals = EntityHashSet::default();
625    for (source, target, weight) in graph.all_edges() {
626        let source = remap.get(&source).copied().unwrap_or(source);
627        let target = remap.get(&target).copied().unwrap_or(target);
628        if let Some(existing_weight) = new_graph.edge_weight(source, target) {
629            if *existing_weight != *weight {
630                removed_intervals.insert(*weight);
631            }
632            continue;
633        }
634        new_graph.add_edge(source, target, *weight);
635    }
636    for node in graph.nodes() {
637        let node = remap.get(&node).copied().unwrap_or(node);
638        new_graph.add_node(node);
639    }
640    graph.map = new_graph;
641
642    for (entry, stop) in &entry_stops {
643        if let Some(new_stop) = remap.get(&stop.0) {
644            commands.entity(entry).insert(EntryStop(*new_stop));
645        }
646    }
647
648    for mut route in &mut routes {
649        for stop in &mut route.stops {
650            if let Some(new_stop) = remap.get(stop) {
651                *stop = *new_stop;
652            }
653        }
654    }
655
656    for interval in removed_intervals {
657        commands.entity(interval).despawn();
658    }
659
660    for duplicate in remap.into_keys() {
661        commands.entity(duplicate).despawn();
662    }
