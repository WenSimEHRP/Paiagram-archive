14impl PaiagramApp {
15    fn new(cc: &eframe::CreationContext) -> Self {
16        cc.egui_ctx.global_style_mut(|style| {
17            style.spacing.window_margin = egui::Margin::same(2);
18            style.interaction.selectable_labels = false;
19        });
20        paiagram_ui::apply_custom_fonts(&cc.egui_ctx);
21        if let Some(render_state) = cc.wgpu_render_state.as_ref() {
22            cc.egui_ctx.data_mut(|data| {
23                data.insert_temp(
24                    egui::Id::new("wgpu_adapter_info"),
25                    eframe::egui_wgpu::adapter_info_summary(&render_state.adapter.get_info()),
26                );
27                data.insert_temp(
28                    egui::Id::new("wgpu_target_format"),
29                    render_state.target_format,
30                );
31                let msaa_samples = if cfg!(target_arch = "wasm32") {
32                    1_u32
33                } else {
34                    4_u32
35                };
36                data.insert_temp(egui::Id::new("wgpu_msaa_samples"), msaa_samples);
37            });
38        }
39        let mut app = App::new();
40        app.add_plugins(MinimalPlugins);
41        app.add_plugins(LogPlugin::default());
42        app.add_plugins((
43            paiagram_ui::UiPlugin,
44            entry::EntryPlugin,
45            graph::GraphPlugin,
46            route::RoutePlugin,
47            import::ImportPlugin,
48            trip::TripPlugin,
49            station::StationPlugin,
50            settings::SettingsPlugin,
51            problems::ProblemsPlugin,
52            class::ClassPlugin,
53            ReadPlugin,
54            SavePlugin,
55            bevy::asset::AssetPlugin::default(),
56            bevy::scene::ScenePlugin,
57        ));
58        info!("Initialized Bevy App.");
59        #[cfg(not(target_arch = "wasm32"))]
60        let args = Arguments::parse();
61        #[cfg(target_arch = "wasm32")]
62        let args = parse_web_arguments();
63        if let Err(e) = app.world_mut().run_system_once_with(handle_args, args) {
64            error!("Failed to web arguments: {:?}", e);
65        } else {
66            info!("Arguments handled successfully.");
67        }
68        Self { bevy_app: app }
69    }

73fn parse_web_arguments() -> Arguments {
74    if let Some(search) =
75        eframe::web_sys::window().and_then(|window| window.location().search().ok())
76    {
77        info!(?search, "Handling web args...");
78        let query = search.strip_prefix('?').unwrap_or(&search);
79        match serde_html_form::from_str::<Arguments>(&query) {
80            Ok(args) => return args,
81            Err(error) => error!("Failed to parse web args: {error}"),
82        }
83    }
84
85    Arguments::default()
86}

98/// Arguments for the application.
99#[derive(Parser, Default, Deserialize)]
100#[command(version, about, long_about = None)]
101struct Arguments {
102    #[arg(
103        short = 'o',
104        long = "open",
105        help = "Path to a .paiagram file (or any other compatible file formats) to open on startup",
106        num_args = 1..
107    )]
108    #[serde(default)]
109    open: Option<Vec<PathBuf>>,
110    #[arg(long = "locale", help = "Set the localization")]
111    #[serde(default)]
112    locale: Option<String>,
113}
114
115fn handle_args(
116    args: In<Arguments>,
117    mut commands: Commands,
118    mut languages: ResMut<UserPreferences>,
119) {
120    for path in args.open.iter().flatten() {
121        #[cfg(target_arch = "wasm32")]
122        {
123            commands.trigger(import::DownloadFile {
124                url: path.to_string_lossy().into_owned(),
125            });
126            continue;
127        }
128
129        #[cfg(not(target_arch = "wasm32"))]
130        {
131            let content = match std::fs::read(path) {
132                Ok(c) => c,
133                Err(e) => {
134                    error!("Could not open {:?}: {:?}", path, e);
135                    continue;
136                }
137            };
138            if let Err(e) = import::load_and_trigger(path, content, &mut commands) {
139                error!("Could not load {:?}: {:#?}", path, e);
140                continue;
141            }
142        }
143    }
144    if let Some(locale) = args.locale.as_deref() {
145        if languages.lang.set(locale) {
146            info!("Successfully set language {}", locale);
147        } else {
148            error!("Unknown language identifier: {}", locale);
149        }
150    }
151}

186#[cfg(target_arch = "wasm32")]
187fn main() {
188    i18n::init();
189    use eframe::wasm_bindgen::JsCast as _;
190    use eframe::web_sys;
191
192    let web_options = eframe::WebOptions::default();
193
194    wasm_bindgen_futures::spawn_local(async {
195        let document = web_sys::window()
196            .expect("No window")
197            .document()
198            .expect("No document");
199
200        let canvas = if let Some(canvas) = document.get_element_by_id("paiagram_canvas") {
201            canvas
202                .dyn_into::<web_sys::HtmlCanvasElement>()
203                .expect("paiagram_canvas was not a HtmlCanvasElement")
204        } else {
205            let canvas = document
206                .create_element("canvas")
207                .expect("Failed to create canvas element");
208            canvas.set_id("paiagram_canvas");
209
210            canvas
211                .set_attribute("style", "display: block; width: 100%; height: 100%;")
212                .ok();
213
214            let body = document.body().expect("Failed to get document body");
215            body.set_attribute(
216                "style",
217                "margin: 0; padding: 0; width: 100%; height: 100%; overflow: hidden;",
218            )
219            .ok();
220
221            let html = document.document_element().expect("No document element");
222            html.set_attribute(
223                "style",
224                "margin: 0; padding: 0; width: 100%; height: 100%; overflow: hidden;",
225            )
226            .ok();
227
228            body.append_child(&canvas).expect("Failed to append canvas");
229            canvas
230                .dyn_into::<web_sys::HtmlCanvasElement>()
231                .expect("Failed to cast canvas")
232        };
233
234        let start_result = eframe::WebRunner::new()
235            .start(
236                canvas,
237                web_options,
238                Box::new(|cc| Ok(Box::new(PaiagramApp::new(cc)))),
239            )
240            .await;
241
242        if let Some(loading_text) = document.get_element_by_id("loading_text") {
243            match start_result {
244                Ok(_) => {
245                    loading_text.remove();
246                }
247                Err(e) => {
248                    loading_text.set_inner_html(
249                        "<p> The app has crashed. See the developer console for details. </p>",
250                    );
251                    panic!("Failed to start eframe: {e:?}");
252                }
253            }
254        }
255    });
