2use bevy::{ecs::entity::EntityHashMap, prelude::*};
3use paiagram_core::{
4    entry::{EntryEstimate, EntryQuery},
5    route::{Route, RouteTrips},
6    station::ParentStationOrStation,
7    trip::TripQuery,
8};
9use smallvec::SmallVec;
10use vec1::{Vec1, vec1};
11
12pub(crate) fn calc(
13    (In(route_entity), InRef(heights), InMut(map)): (
14        In<Entity>,
15        InRef<[(Entity, f32)]>,
16        InMut<Option<TripCache>>,
17    ),
18    route_q: Query<(&RouteTrips, Ref<Route>)>,
19    trip_q: Query<TripQuery>,
20    changed_entries: Query<&ChildOf, Changed<EntryEstimate>>,
21    entries: Query<EntryQuery>,
22    parent_station_or_station: Query<ParentStationOrStation>,
23    mut invalidate_cache: Local<Vec<Entity>>,
24) -> bool {
25    let (trips, route) = route_q.get(route_entity).unwrap();
26
27    let refresh_candidates = if map.is_none() || route.is_changed() {
28        trips.as_slice()
29    } else {
30        if changed_entries.is_empty() {
31            return false;
32        }
33        let mut trips = trips.to_vec();
34        trips.sort_unstable();
35        invalidate_cache.clear();
36        for entity in changed_entries.iter().map(ChildOf::parent) {
37            let Ok(_) = trips.binary_search(&entity) else {
38                continue;
39            };
40            invalidate_cache.push(entity);
41        }
42        invalidate_cache.sort_unstable();
43        invalidate_cache.dedup();
44        invalidate_cache.as_slice()
45    };
46
47    let map: &mut EntityHashMap<SmallVec<[Vec1<TripPoint>; 1]>> = map.get_or_insert_default();
48
49    #[derive(Clone, Copy)]
50    struct TripEntryData {
51        entity: Entity,
52        station: Entity,
53        estimate: Option<EntryEstimate>,
54    }
55
56    struct TripData {
57        entity: Entity,
58        // stroke: DisplayedStroke,
59        entries: Vec<TripEntryData>,
60    }
61
62    let stations_for_layout = &heights[..];
63    let mut station_index_map: EntityHashMap<SmallVec<[usize; 2]>> = EntityHashMap::new();
64    for (idx, (station, _)) in stations_for_layout.iter().enumerate() {
65        station_index_map.entry(*station).or_default().push(idx);
66    }
67
68    let mut trip_data = Vec::with_capacity(refresh_candidates.len());
69    for trip_entity in refresh_candidates.iter().copied() {
70        let trip = trip_q.get(trip_entity).unwrap();
71        let mut trip_entries_vec: Vec<TripEntryData> = Vec::new();
72        for entry_entity in trip.schedule.iter().copied() {
73            let entry = entries.get(entry_entity).unwrap();
74            let station_entity = parent_station_or_station
75                .get(entry.stop())
76                .unwrap()
77                .parent();
78            trip_entries_vec.push(TripEntryData {
79                entity: entry_entity,
80                station: station_entity,
81                estimate: entry.estimate.copied(),
82            });
83        }
84
85        if trip_entries_vec.len() < 2 {
86            continue;
87        }
88
89        trip_data.push(TripData {
90            entity: trip_entity,
91            entries: trip_entries_vec,
92        });
93    }
94
95    for trip in trip_data.into_iter() {
96        let trip_bucket = map.entry(trip.entity).or_default();
97        trip_bucket.clear();
98
99        let mut push_to_bucket = |points: Vec1<TripPoint>| {
100            if points.len() < 2 {
101                return;
102            }
103            trip_bucket.push(points);
104        };
105
106        let trip_entries = trip.entries.as_slice();
107        if trip_entries.len() < 2 {
108            continue;
109        }
110
111        // points, end index
112        let mut local_edges: Vec<Vec1<TripPoint>> = Vec::new();
113        let mut previous_indices: &[usize] = &[];
114
115        if let Some(first) = trip_entries.first()
116            && let Some(indices) = station_index_map.get(&first.station)
117        {
118            previous_indices = indices.as_slice();
119        }
120
121        for entry_idx in 0..trip_entries.len() {
122            let curr = &trip_entries[entry_idx];
123            let next = trip_entries.get(entry_idx + 1);
124
125            if previous_indices.is_empty() {
126                if let Some(next) = next {
127                    if let Some(indices) = station_index_map.get(&next.station) {
128                        previous_indices = indices.as_slice();
129                    } else {
130                        previous_indices = &[];
131                    }
132                }
133                for it in local_edges.drain(..) {
134                    push_to_bucket(it)
135                }
136                continue;
137            }
138
139            let Some(estimate) = curr.estimate else {
140                for it in local_edges.drain(..) {
141                    push_to_bucket(it)
142                }
143                if let Some(next_entry) = next {
144                    if let Some(indices) = station_index_map.get(&next_entry.station) {
145                        previous_indices = indices.as_slice();
146                    } else {
147                        previous_indices = &[];
148                    }
149                }
150                continue;
151            };
152
153            let mut next_local_edges: Vec<Vec1<TripPoint>> = Vec::new();
154
155            for current_line_index in previous_indices.iter().copied() {
156                let matched_idx = local_edges
157                    .iter()
158                    .position(|it| current_line_index.abs_diff(it.last().station_index) <= 1);
159
160                let segment = if let Some(idx) = matched_idx {
161                    let mut a = local_edges.swap_remove(idx);
162                    a.push(TripPoint {
163                        arr: estimate.arr,
164                        dep: estimate.dep,
165                        entry: curr.entity,
166                        station_index: current_line_index,
167                    });
168                    a
169                } else {
170                    vec1![TripPoint {
171                        arr: estimate.arr,
172                        dep: estimate.dep,
173                        entry: curr.entity,
174                        station_index: current_line_index,
175                    }]
176                };
177
178                let mut segment = Some(segment);
179                if let Some(next_entry) = next {
180                    for offset in [-1, 0, 1] {
181                        let next_idx = (current_line_index as isize + offset) as usize;
182                        if let Some((s, _)) = stations_for_layout.get(next_idx) {
183                            if *s == next_entry.station {
184                                let forwarded_segment = segment.take().unwrap();
185                                next_local_edges.push(forwarded_segment);
186                                break;
187                            }
188                        }
189                    }
190                }
191
192                if let Some(it) = segment {
193                    push_to_bucket(it);
194                }
195            }
196
197            for it in local_edges.drain(..) {
198                push_to_bucket(it)
199            }
200
201            local_edges = next_local_edges;
202            if let Some(next_entry) = next {
203                if let Some(indices) = station_index_map.get(&next_entry.station) {
204                    previous_indices = indices.as_slice();
205                } else {
206                    previous_indices = &[];
207                }
208            }
209        }
210
211        for it in local_edges.drain(..) {
212            push_to_bucket(it)
213        }
214    }
215    return true;
216}
