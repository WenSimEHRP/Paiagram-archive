3/// Display the time indicator's indicator given the SCREEN coordinates
4/// of the clip rect and the time indicator.
5pub fn display_time_indicator_indicator_vertical(
6    id: Id,
7    rect: Rect,
8    time_indicator_y: f32,
9    color: Color32,
10    painter: &Painter,
11) {
12    let y_range = rect.y_range();
13    let in_viewport = y_range.contains(time_indicator_y);
14    let view_strength = painter.ctx().animate_bool(id, !in_viewport);
15    let rect = if time_indicator_y < (y_range.min + y_range.max) / 2.0 {
16        Rect::from_two_pos(rect.left_top(), rect.right_top())
17    } else {
18        Rect::from_two_pos(rect.left_bottom(), rect.right_bottom())
19    };
20    let shape = Shadow {
21        offset: [0, 0],
22        blur: 255,
23        spread: (8.0 * view_strength) as u8,
24        color: color.gamma_multiply(0.2).gamma_multiply(view_strength),
25    }
26    .as_shape(rect, 0);
27    painter.add(shape);
28}
