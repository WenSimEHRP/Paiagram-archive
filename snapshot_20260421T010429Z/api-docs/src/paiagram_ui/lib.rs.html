134fn update_selected_items(
135    mut msg: MessageReader<ModifySelectedItems>,
136    mut selected_items: ResMut<SelectedItems>,

141            ModifySelectedItems::SetSingle(it) => selected_items.set_single_selection(it.clone()),

147impl SelectedItems {
148    fn toggle_entries<T: Ord + Clone>(
149        entries: &mut Vec1<T>,
150        incoming: impl Iterator<Item = T>,
151    ) -> bool {
152        // 1. Combine and Sort
153        entries.extend(incoming);
154        entries.sort_unstable();
155
156        let mut result = Vec::new();
157        let mut i = 0;
158        let data = entries.as_vec(); // Access inner vec
159
160        // We look for contiguous groups. If a group has an odd length, we keep 1.
161        // If it's even, we keep 0.
162        while i < data.len() {
163            let mut j = i + 1;
164            while j < data.len() && data[j] == data[i] {
165                j += 1;
166            }
167
168            if (j - i) % 2 != 0 {
169                result.push(data[i].clone());
170            }
171            i = j;
172        }
173
174        if result.is_empty() {
175            true
176        } else {
177            *entries = Vec1::try_from_vec(result).unwrap();
178            false
179        }
180    }

182    pub(crate) fn toggle_selection(&mut self, item: SelectedItem) {
183        let mut should_reset = false;
184
185        match (item, &mut *self) {
186            (SelectedItem::None, _) => return,
187            (SelectedItem::Trip(sel), Self::Trips(it)) => {
188                it.sort_unstable_by_key(|t| t.trip);
189
190                match it.binary_search_by_key(&sel.trip, |t| t.trip) {
191                    // Trip exists, toggle the internal entries
192                    Ok(idx) => {
193                        if Self::toggle_entries(&mut it[idx].entries, sel.entries.into_iter()) {
194                            if it.len() == 1 {
195                                should_reset = true;
196                            } else {
197                                it.remove(idx);
198                            }
199                        }
200                    }
201                    Err(idx) => {
202                        it.insert(idx, sel);
203                    }
204                }
205            }
206
207            (SelectedItem::Interval(sel), Self::Intervals(it)) => {
208                should_reset = Self::toggle_entries(it, std::iter::once(sel));
209            }
210
211            (SelectedItem::Station(sel), Self::Stations(it)) => {
212                should_reset = Self::toggle_entries(it, std::iter::once(sel));
213            }
214            _ => {}
215        }
216
217        if should_reset {
218            *self = SelectedItems::None;
219        }
220    }
221
222    pub(crate) fn set_single_selection(&mut self, item: SelectedItem) {
223        match item {
224            SelectedItem::None => *self = Self::None,
225            SelectedItem::Trip(it) => *self = Self::Trips(vec1![it]),
226            SelectedItem::Interval(it) => *self = Self::Intervals(vec1![it]),
227            SelectedItem::Station(it) => *self = Self::Stations(vec1![it]),
228        }
229    }
230}

282#[derive(Resource)]
283struct FrameTimeHistory {
284    values: [f32; Self::CAPACITY],
285    next_index: usize,
286}
287
288impl FrameTimeHistory {
289    const CAPACITY: usize = 255;
290
291    fn push(&mut self, dt_seconds: f32) {
292        self.values[self.next_index] = dt_seconds;
293        self.next_index = (self.next_index + 1) % Self::CAPACITY;
294    }
295
296    fn average_dt(&self) -> f32 {
297        let sum: f32 = self.values.iter().sum();
298        sum / Self::CAPACITY as f32
299    }
300
301    fn previous_n(&self, n: usize) -> impl Iterator<Item = f32> {
302        let count = n.min(Self::CAPACITY);
303        (0..count).map(move |i| {
304            let index = (self.next_index + Self::CAPACITY - 1 - i) % Self::CAPACITY;
305            self.values[index]
306        })
307    }
308}
309
310impl Default for FrameTimeHistory {
311    fn default() -> Self {
312        Self {
313            values: [0.0; Self::CAPACITY],
314            next_index: 0,
315        }
316    }
317}

319#[derive(Resource, Reflect)]
320#[reflect(Resource)]
321pub struct GlobalTimer {
322    value: AtomicI64,
323    locker: AtomicU64,
324    animation_speed: f64,
325    animation_playing: bool,
326    sync_to_real_time: bool,
327}
328
329fn update_timer(mut timer: ResMut<GlobalTimer>, time: Res<Time<Real>>) {
330    if !timer.is_locked() && timer.sync_to_real_time {
331        let now = Local::now();
332        let seconds = now.num_seconds_from_midnight() as f64;
333        let rest = now.nanosecond() as f64 / 1_000_000_000 as f64;
334        timer.animation_speed = 1.0;
335        timer.animation_playing = true;
336        timer.write_seconds(seconds + rest);
337    } else if timer.animation_playing && !timer.is_locked() {
338        let mut seconds = timer.read_seconds();
339        seconds += timer.animation_speed * time.delta_secs_f64();
340        timer.write_seconds(seconds);
341    }
342}
343
344impl Default for GlobalTimer {
345    fn default() -> Self {
346        Self {
347            value: AtomicI64::new(0),
348            locker: AtomicU64::new(Self::UNLOCKED),
349            animation_speed: 10.0,
350            animation_playing: false,
351            sync_to_real_time: false,
352        }
353    }
354}
355
356impl GlobalTimer {
357    const UNLOCKED: u64 = u64::MAX;
358    pub fn read_ticks(&self) -> Tick {
359        Tick(self.value.load(Ordering::Acquire))
360    }
361
362    pub fn write_ticks(&self, value: Tick) {
363        self.value.store(value.0, Ordering::Release);
364    }
365
366    pub fn read_seconds(&self) -> f64 {
367        self.read_ticks().as_seconds_f64()
368    }
369
370    pub fn write_seconds(&self, value: f64) {
371        let ticks_per_second = Tick::from_timetable_time(TimetableTime(1)).0 as f64;
372        let ticks = (value * ticks_per_second).round() as i64;
373        self.write_ticks(Tick(ticks));
374    }
375
376    pub fn is_locked(&self) -> bool {
377        self.locker.load(Ordering::Acquire) != Self::UNLOCKED
378    }
379
380    pub fn try_lock(&self, id: Entity) -> bool {
381        let id_bits = id.to_bits();
382
383        let result = self.locker.compare_exchange(
384            Self::UNLOCKED,
385            id_bits,
386            Ordering::AcqRel,
387            Ordering::Acquire,
388        );
389
390        result.is_ok() || result.unwrap_err() == id_bits
391    }
392
393    pub fn try_unlock(&self, id: Entity) {
394        let _ = self.locker.compare_exchange(
395            id.to_bits(),
396            Self::UNLOCKED,
397            Ordering::Release,
398            Ordering::Relaxed,
399        );
400    }
401
402    pub fn owner(&self) -> u64 {
403        self.locker.load(Ordering::Acquire)
404    }
405
406    pub unsafe fn try_lock_unchecked(&self, id: u64) -> bool {
407        self.try_lock(Entity::from_bits(id))
408    }
409
410    pub unsafe fn try_unlock_unchecked(&self, id: u64) {
411        self.try_unlock(Entity::from_bits(id))
412    }
413}

477impl MainUiState {
478    pub fn push_to_focused_leaf(&mut self, new_pane: MainTab) -> TileId {
479        let new_id = self.tree.tiles.insert_pane(new_pane);
480
481        // Try to add it to the same Tabs container that is currently focused
482        if let Some(&active_id) = self.tree.active_tiles().last()
483            && let Some(parent_id) = self.tree.tiles.parent_of(active_id)
484            && let Some(Tile::Container(container)) = self.tree.tiles.get_mut(parent_id)
485            && container.kind() == ContainerKind::Tabs
486        {
487            container.add_child(new_id);
488            self.tree.make_active(|id, _| id == new_id);
489            return new_id;
490        }
491
492        // Fallback: create a new top-level Tabs container
493        let old_root = self.tree.root;
494        let tabs_id = if let Some(old_root) = old_root {
495            self.tree.tiles.insert_tab_tile(vec![old_root, new_id])
496        } else {
497            self.tree.tiles.insert_tab_tile(vec![new_id])
498        };
499        self.tree.root = Some(tabs_id);
500        self.tree.make_active(|id, _| id == new_id);
501        new_id
502    }
503}

527#[derive(Message)]
528struct OpenOrFocus(MainTab);
529
530fn open_or_focus_tab(
531    mut messages: MessageReader<OpenOrFocus>,
532    mut mus: ResMut<MainUiState>,
533    mut aus: ResMut<AdditionalUiState>,
534) {
535    for msg in messages.read() {
536        let pane = &msg.0; // your pane data
537
538        let focused_id = if let Some(tile_id) = mus.tree.tiles.find_pane(pane) {
539            // Already exists → just focus it
540            mus.make_active(|id, _| id == tile_id);
541            mus.set_visible(tile_id, true);
542            tile_id
543        } else {
544            // New pane → add it to the currently focused container
545            mus.push_to_focused_leaf(pane.clone())
546        };
547        aus.focused_id = Some(focused_id);
548    }
549}

710        visuals: &egui::Visuals,
711        tiles: &Tiles<MainTab>,
712        tile_id: TileId,
713        state: &egui_tiles::TabState,
714    ) -> Color32 {
715        let base = match tiles.get(tile_id) {
716            None | Some(Tile::Container(_)) => visuals.panel_fill,
717            Some(Tile::Pane(tab)) => {
718                DisplayedColor::from_seed(for_all_tab_types!(tab, NAME)).get(visuals.dark_mode)
719            }
720        };
721        base.gamma_multiply(if state.active { 0.7 } else { 0.2 })
722    }
723    fn tab_outline_stroke(
724        &self,
725        visuals: &egui::Visuals,
726        tiles: &Tiles<MainTab>,
727        tile_id: TileId,
728        state: &egui_tiles::TabState,
729    ) -> Stroke {
730        let base = match tiles.get(tile_id) {
731            None | Some(Tile::Container(_)) => visuals.panel_fill,
732            Some(Tile::Pane(tab)) => {
733                DisplayedColor::from_seed(for_all_tab_types!(tab, NAME)).get(visuals.dark_mode)
734            }
735        };
736        Stroke::new(
737            1.0,
738            base.gamma_multiply(if state.active { 1.0 } else { 0.7 }),
739        )
740    }
741    fn tab_bar_hline_stroke(&self, visuals: &egui::Visuals) -> Stroke {
742        Stroke::new(1.0, Color32::TRANSPARENT)
743    }

1157fn build_font_definitions(sarasa: Option<Vec<u8>>) -> egui::FontDefinitions {
1158    let mut fonts = egui::FontDefinitions::default();
1159
1160    let has_sarasa = sarasa.is_some();
1161    if let Some(bytes) = sarasa {
1162        fonts.font_data.insert(
1163            "my_font".to_owned(),
1164            std::sync::Arc::new(egui::FontData::from_owned(bytes)),
1165        );
1166    }
1167
1168    fonts.font_data.insert(
1169        "dia_pro".to_owned(),
1170        std::sync::Arc::new(egui::FontData::from_static(include_bytes!(
1171            "../assets/fonts/DiaPro-Regular.ttf"
1172        ))),
1173    );
1174
1175    if has_sarasa {
1176        fonts
1177            .families
1178            .get_mut(&egui::FontFamily::Proportional)
1179            .unwrap()
1180            .insert(0, "my_font".to_owned());
1181    }
1182
1183    let mut dia_pro_family = vec!["dia_pro".to_owned()];
1184    if has_sarasa {
1185        dia_pro_family.push("my_font".to_owned());
1186    }
1187    fonts
1188        .families
1189        .insert(egui::FontFamily::Name("dia_pro".into()), dia_pro_family);
1190
1191    fonts
1192}

1194#[cfg(not(target_arch = "wasm32"))]
1195fn load_sarasa_local() -> Option<Vec<u8>> {
1196    let mut candidates = vec![
1197        PathBuf::from("assets/fonts/SarasaUiSC-Regular.ttf"),
1198        PathBuf::from("crates/paiagram-ui/assets/fonts/SarasaUiSC-Regular.ttf"),
1199    ];
1200
1201    if let Ok(exe) = std::env::current_exe()
1202        && let Some(parent) = exe.parent()
1203    {
1204        candidates.push(parent.join("assets/fonts/SarasaUiSC-Regular.ttf"));
1205    }
1206
1207    for path in candidates {
1208        if let Ok(bytes) = std::fs::read(&path) {
1209            return Some(bytes);
1210        }
1211    }
1212
1213    None
1214}

1216#[cfg(target_arch = "wasm32")]
1217fn download_sarasa_and_apply(ctx: Context) {
1218    wasm_bindgen_futures::spawn_local(async move {
1219        let Some(window) = eframe::web_sys::window() else {
1220            return;
1221        };
1222
1223        let Ok(response) =
1224            wasm_bindgen_futures::JsFuture::from(window.fetch_with_str("SarasaUiSC-Regular.ttf"))
1225                .await
1226        else {
1227            return;
1228        };
1229
1230        let Ok(response) = response.dyn_into::<eframe::web_sys::Response>() else {
1231            return;
1232        };
1233
1234        if !response.ok() {
1235            return;
1236        }
1237
1238        let Ok(array_buffer_promise) = response.array_buffer() else {
1239            return;
1240        };
1241
1242        let Ok(array_buffer) = wasm_bindgen_futures::JsFuture::from(array_buffer_promise).await
1243        else {
1244            return;
1245        };
1246
1247        let bytes = js_sys::Uint8Array::new(&array_buffer).to_vec();
1248        ctx.set_fonts(build_font_definitions(Some(bytes)));
1249    });
