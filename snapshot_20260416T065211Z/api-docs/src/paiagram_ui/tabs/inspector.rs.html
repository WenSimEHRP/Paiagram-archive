19struct EntityTable<'w> {
20    name_or_entity_iter: QueryIter<'w, 'w, NameOrEntity, Without<EntryMode>>,
21    names_or_entities: Vec<NameOrEntityItem<'w, 'w>>,
22    selected: &'w mut SelectedItem,
23    start: usize,
24}
25
26impl<'w> TableDelegate for EntityTable<'w> {
27    fn header_cell_ui(&mut self, _ui: &mut egui::Ui, _cell: &egui_table::HeaderCellInfo) {}
28    fn prepare(&mut self, info: &egui_table::PrefetchInfo) {
29        let start = info.visible_rows.start as usize;
30        let end = info.visible_rows.end as usize;
31        self.start = start;
32        let count = end.saturating_sub(start).saturating_add(1);
33        self.names_or_entities
34            .extend(self.name_or_entity_iter.clone().skip(start).take(count));
35    }
36    fn cell_ui(&mut self, ui: &mut egui::Ui, cell: &egui_table::CellInfo) {
37        let info = &self.names_or_entities[cell.row_nr as usize - self.start];
38        ui.selectable_value(
39            self.selected,
40            SelectedItem::Entity(info.entity),
41            if let Some(n) = info.name {
42                format!("{} ({})", n, info.entity)
43            } else {
44                info.entity.to_string()
45            },
46        );
47    }
48}
