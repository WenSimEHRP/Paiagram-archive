14#[derive(Serialize, Deserialize, Clone, Default, MapEntities, PartialEq)]
15pub struct SettingsTab;
16
17impl Tab for SettingsTab {
18    const NAME: &'static str = "Settings";
19    fn main_display(&mut self, world: &mut World, ui: &mut Ui) {
20        if let Err(e) = world.run_system_cached_with(show_settings, ui) {
21            bevy::log::error!("UI Error while displaying settings page: {}", e)
22        }
23    }
24    fn title(&self) -> egui::WidgetText {
25        tr!("tab-settings").into()
26    }
27}
28
29fn show_settings(
30    InMut(ui): InMut<Ui>,
31    mut preferences: ResMut<UserPreferences>,
32    mut settings: ResMut<ProjectSettings>,
33) {
34    ui.heading(tr!("settings-preferences"));
35    egui::Grid::new("settings grid 1").show(ui, |ui| {
36        ui.label(tr!("settings-dark-mode"));
37        ui.checkbox(&mut preferences.dark_mode, "");
38        ui.end_row();
39
40        ui.label(tr!("settings-language"));
41        egui::ComboBox::new("language", "")
42            .selected_text(preferences.lang.name())
43            .show_ui(ui, |ui| {
44                for lang in Language::iter() {
45                    ui.selectable_value(&mut preferences.lang, lang, lang.name());
46                }
47            });
48        ui.end_row();
49
50        ui.label("Developer Mode");
51        ui.checkbox(&mut preferences.developer_mode, "");
52        ui.end_row();
53
54        ui.label("Antialising Options");
55        egui::ComboBox::new("antialiasing", "")
56            .selected_text(match preferences.antialiasing_mode {
57                AntialiasingMode::Off => "Off",
58                AntialiasingMode::On => "On",
59            })
60            .show_ui(ui, |ui| {
61                ui.selectable_value(
62                    &mut preferences.antialiasing_mode,
63                    AntialiasingMode::Off,
64                    "Off",
65                );
66                ui.selectable_value(
67                    &mut preferences.antialiasing_mode,
68                    AntialiasingMode::On,
69                    "On",
70                );
71            });
72        ui.end_row();
73
74        ui.label("LOD Mode");
75        egui::ComboBox::new("lod", "")
76            .selected_text(match preferences.level_of_detail_mode {
77                LevelOfDetailMode::Off => "Off",
78                LevelOfDetailMode::Lod2 => "2×",
79                LevelOfDetailMode::Lod4 => "4×",
80            })
81            .show_ui(ui, |ui| {
82                ui.selectable_value(
83                    &mut preferences.level_of_detail_mode,
84                    LevelOfDetailMode::Off,
85                    "Off",
86                );
87                ui.selectable_value(
88                    &mut preferences.level_of_detail_mode,
89                    LevelOfDetailMode::Lod2,
90                    "2×",
91                );
92                ui.selectable_value(
93                    &mut preferences.level_of_detail_mode,
94                    LevelOfDetailMode::Lod4,
95                    "4×",
96                );
97            });
98        ui.end_row();
99    });
100    ui.heading(tr!("settings-project-settings"));
101    ui.text_edit_multiline(&mut settings.remarks);
102}
